53  class ImportTest {
54  
55    @NonNull
56    private static IDocumentNodeItem newImportedCatalog() {
57  
58      // setup the imported catalog
59      Catalog importedCatalog = new Catalog();
60  
61      importedCatalog.addControl(AbstractControl.builder("control1")
62          .title("Control 1")
63          .build());
64      importedCatalog.addControl(AbstractControl.builder("control2")
65          .title("Control 2")
66          .build());
67  
68      return DefaultNodeItemFactory.instance().newDocumentNodeItem(
69          IRootAssemblyDefinition.toRootAssemblyDefinition(
70              ObjectUtils.notNull(
71                  (IAssemblyClassBinding) OscalBindingContext.instance().getClassBinding(Catalog.class))),
72          importedCatalog,
73          ObjectUtils.notNull(Paths.get("").toUri()));
74    }
75  
76    @SuppressWarnings("null")
77    @Test
78    void test() throws ProfileResolutionException {
79      URI cwd = Paths.get("").toUri();
80  
81      // setup the imported catalog
82      IDocumentNodeItem importedCatalogDocumentItem = newImportedCatalog();
83  
84      // setup the profile
85      Profile profile = new Profile();
86  
87      ProfileImport profileImport = new ProfileImport();
88      profileImport.setIncludeAll(new IncludeAll());
89      profileImport.setExcludeControls(Collections.singletonList(
90          AbstractProfileSelectControlById.builder()
91              .withId("control1")
92              .build()));
93      profileImport.setHref(cwd);
94      profile.addImport(profileImport);
95  
96      IDocumentNodeItem profileDocumentItem = DefaultNodeItemFactory.instance().newDocumentNodeItem(
97          IRootAssemblyDefinition.toRootAssemblyDefinition(
98              ObjectUtils.notNull(
99                  (IAssemblyClassBinding) OscalBindingContext.instance().getClassBinding(Profile.class))),
100         profile,
101         cwd);
102 
103     // setup the resolved catalog
104     Catalog resolvedCatalog = new Catalog();
105 
106     for (IRequiredValueModelNodeItem importItem : CollectionUtil.toIterable(
107         profileDocumentItem.getModelItemsByName("profile").stream()
108             .flatMap(root -> root.getModelItemsByName("import").stream()))) {
109 
110       Import catalogImport = new Import(profileDocumentItem, importItem);
111       catalogImport.resolve(importedCatalogDocumentItem, resolvedCatalog);
112     }
113   }
