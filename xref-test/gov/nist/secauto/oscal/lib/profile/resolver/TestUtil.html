53    @NonNull
54    public static final IIdentifierMapper UUID_CONCAT_ID_MAPPER = new IIdentifierMapper() {
55      private final Map<String, String> idToReassignmentMap = new ConcurrentHashMap<>();
56  
57      @NonNull
58      private String reassign(@NonNull String identifier) {
59        String retval = idToReassignmentMap.get(identifier);
60        if (retval == null) {
61          retval = identifier + "-" + UUID.randomUUID().toString();
62          idToReassignmentMap.put(identifier, retval);
63        }
64        return retval;
65      }
66  
67      @Override
68      public String mapRoleIdentifier(@NonNull String identifier) {
69        return reassign(identifier);
70      }
71  
72      @Override
73      public String mapControlIdentifier(@NonNull String identifier) {
74        return reassign(identifier);
75      }
76  
77      @Override
78      public String mapGroupIdentifier(@NonNull String identifier) {
79        return reassign(identifier);
80      }
81  
82      @Override
83      public String mapParameterIdentifier(@NonNull String identifier) {
84        return reassign(identifier);
85      }
86  
87      @Override
88      public @NonNull String mapPartIdentifier(@NonNull String identifier) {
89        return reassign(identifier);
90      }
91    };
