027package gov.nist.secauto.oscal.lib.profile.resolver.policy;
028
029import gov.nist.secauto.metaschema.model.common.metapath.item.IRequiredValueModelNodeItem;
030import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
031import gov.nist.secauto.oscal.lib.profile.resolver.ProfileResolutionEvaluationException;
032import gov.nist.secauto.oscal.lib.profile.resolver.support.IEntityItem;
033
034import org.apache.logging.log4j.LogManager;
035import org.apache.logging.log4j.Logger;
036
037import java.util.List;
038
039import edu.umd.cs.findbugs.annotations.NonNull;
040import edu.umd.cs.findbugs.annotations.Nullable;
041
042public abstract class AbstractCustomReferencePolicy<TYPE> implements ICustomReferencePolicy<TYPE> {
043  private static final Logger LOGGER = LogManager.getLogger(AbstractCustomReferencePolicy.class);
044
045  @NonNull
046  private final IIdentifierParser identifierParser;
047
048  protected AbstractCustomReferencePolicy(
049      @NonNull IIdentifierParser identifierParser) {
050    this.identifierParser = identifierParser;
051  }
052
053  @Override
054  @NonNull
055  public IIdentifierParser getIdentifierParser() {
056    return identifierParser;
057  }
058
059  /**
060   * Get the possible item types that can be searched in the order in which the identifier will be
061   * looked up.
062   * <p>
063   * The {@code reference} object is provided to allow for context sensitive item type tailoring.
064   *
065   * @param reference
066   *          the reference object
067   * @return a list of item types to search for
068   */
069  @NonNull
070  protected abstract List<IEntityItem.ItemType> getEntityItemTypes(@NonNull TYPE reference);
071
072  /**
073   * Handle an index hit.
074   *
075   * @param contextItem
076   *          the node containing the identifier reference
077   * @param reference
078   *          the identifier reference object generating the hit
079   * @param item
080   *          the referenced item
081   * @param visitorContext
082   *          the reference visitor state, which can be used for further processing
083   * @return {@code true} if the hit was handled or {@code false} otherwise
084   * @throws ProfileResolutionEvaluationException
085   *           if there was an error handing the index hit
086   */
087  protected boolean handleIndexHit(
088      @NonNull IRequiredValueModelNodeItem contextItem,
089      @NonNull TYPE reference,
090      @NonNull IEntityItem item,
091      @NonNull ReferenceCountingVisitor.Context visitorContext) {
092
093    if (visitorContext.getIndexer().isSelected(item)) {
094      if (!visitorContext.isResolved(item)) {
095        // this referenced item will need to be resolved
096        ReferenceCountingVisitor.instance().resolveEntity(item, visitorContext);
097      }
098      item.incrementReferenceCount();
099
100      if (item.isIdentifierReassigned()) {
101        String referenceText = ObjectUtils.notNull(getReferenceText(reference));
102        String newReferenceText = getIdentifierParser().update(referenceText, item.getIdentifier());
103        setReferenceText(reference, newReferenceText);
104        if (LOGGER.isDebugEnabled()) {
105          LOGGER.atDebug().log("Mapping {} reference '{}' to '{}'.", item.getItemType().name(), referenceText,
106              newReferenceText);
107        }
108      }
109      handleSelected(contextItem, reference, item, visitorContext);
110    } else {
111      handleUnselected(contextItem, reference, item, visitorContext);
112    }
113    return true;
114  }
115
116  /**
117   * Handle an index hit against an item related to an unselected control.
118   * <p>
119   * Subclasses can override this method to perform extra processing.
120   *
121   * @param contextItem
122   *          the node containing the identifier reference
123   * @param reference
124   *          the identifier reference object generating the hit
125   * @param item
126   *          the referenced item
127   * @param visitorContext
128   *          the reference visitor, which can be used for further processing
129   * @throws ProfileResolutionEvaluationException
130   *           if there was an error handing the index hit
131   */
132  protected void handleUnselected( // NOPMD noop default
133      @NonNull IRequiredValueModelNodeItem contextItem,
134      @NonNull TYPE reference,
135      @NonNull IEntityItem item,
136      @NonNull ReferenceCountingVisitor.Context visitorContext) {
137    // do nothing by default
138  }
139
140  /**
141   * Handle an index hit against an item related to an selected control.
142   * <p>
143   * Subclasses can override this method to perform extra processing.
144   *
145   * @param contextItem
146   *          the node containing the identifier reference
147   * @param reference
148   *          the identifier reference object generating the hit
149   * @param item
150   *          the referenced item
151   * @param visitorContext
152   *          the reference visitor state, which can be used for further processing
153   * @throws ProfileResolutionEvaluationException
154   *           if there was an error handing the index hit
155   */
156  protected void handleSelected( // NOPMD noop default
157      @NonNull IRequiredValueModelNodeItem contextItem,
158      @NonNull TYPE reference,
159      @NonNull IEntityItem item,
160      @NonNull ReferenceCountingVisitor.Context visitorContext) {
161    // do nothing by default
162  }
163
164  /**
165   * Handle an index miss for a reference. This occurs when the referenced item was not found in the
166   * index.
167   * <p>
168   * Subclasses can override this method to perform extra processing.
169   *
170   * @param contextItem
171   *          the node containing the identifier reference
172   * @param reference
173   *          the identifier reference object generating the hit
174   * @param itemTypes
175   *          the possible item types for this reference
176   * @param identifier
177   *          the parsed identifier
178   * @param visitorContext
179   *          the reference visitor state, which can be used for further processing
180   * @return {@code true} if the reference is handled by this method or {@code false} otherwise
181   * @throws ProfileResolutionEvaluationException
182   *           if there was an error handing the index miss
183   */
184  protected boolean handleIndexMiss(
185      @NonNull IRequiredValueModelNodeItem contextItem,
186      @NonNull TYPE reference,
187      @NonNull List<IEntityItem.ItemType> itemTypes,
188      @NonNull String identifier,
189      @NonNull ReferenceCountingVisitor.Context visitorContext) {
190    // provide no handler by default
191    return false;
192  }
193
194  /**
195   * Handle the case where the identifier was not a syntax match for an expected identifier. This can
196   * occur when the reference is malformed, using an unrecognized syntax.
197   * <p>
198   * Subclasses can override this method to perform extra processing.
199   *
200   * @param contextItem
201   *          the node containing the identifier reference
202   * @param reference
203   *          the identifier reference object generating the hit
204   * @param visitorContext
205   *          the reference visitor state, which can be used for further processing
206   * @return {@code true} if the reference is handled by this method or {@code false} otherwise
207   * @throws ProfileResolutionEvaluationException
208   *           if there was an error handing the index miss due to a non match
209   */
210  protected boolean handleIdentifierNonMatch(
211      @NonNull IRequiredValueModelNodeItem contextItem,
212      @NonNull TYPE reference,
213      @NonNull ReferenceCountingVisitor.Context visitorContext) {
214    // provide no handler by default
215    return false;
216  }
217
218  @Override
219  public boolean handleReference(
220      @NonNull IRequiredValueModelNodeItem contextItem,
221      @NonNull TYPE type,
222      @NonNull ReferenceCountingVisitor.Context visitorContext) {
223    String referenceText = getReferenceText(type);
224
225    // if the reference text does not exist, ignore the reference; otherwise, handle it.
226    return referenceText == null
227        || handleIdentifier(contextItem, type, getIdentifierParser().parse(referenceText), visitorContext);
228  }
229
230  /**
231   * Handle the provided {@code identifier} for a given {@code type} of reference.
232   *
233   * @param contextItem
234   *          the node containing the identifier reference
235   * @param type
236   *          the item type of the reference
237   * @param identifier
238   *          the identifier
239   * @param visitorContext
240   *          the reference visitor state, which can be used for further processing
241   * @return {@code true} if the reference is handled by this method or {@code false} otherwise
242   * @throws ProfileResolutionEvaluationException
243   *           if there was an error handing the reference
244   */
245  protected boolean handleIdentifier(
246      @NonNull IRequiredValueModelNodeItem contextItem,
247      @NonNull TYPE type,
248      @Nullable String identifier,
249      @NonNull ReferenceCountingVisitor.Context visitorContext) {
250    boolean retval;
251    if (identifier == null) {
252      retval = handleIdentifierNonMatch(contextItem, type, visitorContext);
253    } else {
254      List<IEntityItem.ItemType> itemTypes = getEntityItemTypes(type);
255      IEntityItem item = null;
256      for (IEntityItem.ItemType itemType : itemTypes) {
257        assert itemType != null;
258
259        item = visitorContext.getEntity(itemType, identifier);
260        if (item != null) {
261          break;
262        }
263      }
264
265      if (item == null) {
266        retval = handleIndexMiss(contextItem, type, itemTypes, identifier, visitorContext);
267      } else {
268        retval = handleIndexHit(contextItem, type, item, visitorContext);
269      }
270    }
271    return retval;
272  }
