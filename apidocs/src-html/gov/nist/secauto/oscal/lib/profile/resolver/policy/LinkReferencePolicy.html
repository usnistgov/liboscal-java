027package gov.nist.secauto.oscal.lib.profile.resolver.policy;
028
029import gov.nist.secauto.metaschema.model.common.metapath.format.IPathFormatter;
030import gov.nist.secauto.metaschema.model.common.metapath.item.IRequiredValueModelNodeItem;
031import gov.nist.secauto.metaschema.model.common.util.CollectionUtil;
032import gov.nist.secauto.metaschema.model.common.util.CustomCollectors;
033import gov.nist.secauto.oscal.lib.model.Link;
034import gov.nist.secauto.oscal.lib.profile.resolver.support.IEntityItem;
035
036import org.apache.logging.log4j.LogManager;
037import org.apache.logging.log4j.Logger;
038
039import java.net.URI;
040import java.util.List;
041import java.util.Locale;
042
043import edu.umd.cs.findbugs.annotations.NonNull;
044
045public class LinkReferencePolicy
046    extends AbstractMultiItemTypeReferencePolicy<Link> {
047  private static final Logger LOGGER = LogManager.getLogger(LinkReferencePolicy.class);
048
049  @SuppressWarnings("null")
050  @NonNull
051  public static LinkReferencePolicy create(@NonNull IEntityItem.ItemType itemType) {
052    return create(List.of(itemType));
053  }
054
055  @NonNull
056  public static LinkReferencePolicy create(@NonNull List<IEntityItem.ItemType> itemTypes) {
057    return new LinkReferencePolicy(CollectionUtil.requireNonEmpty(itemTypes, "itemTypes"));
058  }
059
060  public LinkReferencePolicy(@NonNull List<IEntityItem.ItemType> itemTypes) {
061    super(IIdentifierParser.FRAGMENT_PARSER, itemTypes);
062  }
063
064  @Override
065  public String getReferenceText(@NonNull Link link) {
066    return link.getHref().toString();
067  }
068
069  @Override
070  public void setReferenceText(@NonNull Link link, @NonNull String newValue) {
071    link.setHref(URI.create(newValue));
072  }
073
074  @Override
075  protected void handleUnselected(
076      @NonNull IRequiredValueModelNodeItem contextItem,
077      @NonNull Link link,
078      @NonNull IEntityItem item,
079      @NonNull ReferenceCountingVisitor.Context visitorContext) {
080    URI linkHref = link.getHref();
081    URI sourceUri = item.getSource();
082
083    URI resolved = sourceUri.resolve(linkHref);
084    if (LOGGER.isTraceEnabled()) {
085      LOGGER.atTrace().log("At path '{}', remapping orphaned URI '{}' to '{}'",
086          contextItem.toPath(IPathFormatter.METAPATH_PATH_FORMATER),
087          linkHref.toString(),
088          resolved.toString());
089    }
090    link.setHref(resolved);
091  }
092
093  @Override
094  protected boolean handleIndexMiss(
095      @NonNull IRequiredValueModelNodeItem contextItem,
096      @NonNull Link link,
097      @NonNull List<IEntityItem.ItemType> itemTypes,
098      @NonNull String identifier,
099      @NonNull ReferenceCountingVisitor.Context visitorContext) {
100    if (LOGGER.isWarnEnabled()) {
101      LOGGER.atWarn().log(
102          "The link at '{}' with rel '{}' should reference a {} identified by '{}'."
103              + " The index did not contain the identifier.",
104          contextItem.toPath(IPathFormatter.METAPATH_PATH_FORMATER),
105          link.getRel(),
106          itemTypes.stream()
107              .map(en -> en.name().toLowerCase(Locale.ROOT))
108              .collect(CustomCollectors.joiningWithOxfordComma("or")),
109          identifier);
110    }
111    return true;
112  }
113
114  @Override
115  protected boolean handleIdentifierNonMatch(
116      @NonNull IRequiredValueModelNodeItem contextItem,
117      @NonNull Link reference,
118      @NonNull ReferenceCountingVisitor.Context visitorContext) {
119    if (LOGGER.isDebugEnabled()) {
120      LOGGER.atDebug().log("Ignoring URI '{}' at '{}'",
121          reference.getHref().toString(),
122          contextItem.toPath(IPathFormatter.METAPATH_PATH_FORMATER));
123    }
124
125    return true;
126  }
127}
