027package gov.nist.secauto.oscal.lib.profile.resolver.policy;
028
029import com.vladsch.flexmark.util.sequence.BasedSequence;
030
031import gov.nist.secauto.metaschema.model.common.datatype.markup.flexmark.InsertAnchorExtension.InsertAnchorNode;
032import gov.nist.secauto.metaschema.model.common.metapath.format.IPathFormatter;
033import gov.nist.secauto.metaschema.model.common.metapath.item.IRequiredValueModelNodeItem;
034import gov.nist.secauto.metaschema.model.common.util.CollectionUtil;
035import gov.nist.secauto.metaschema.model.common.util.CustomCollectors;
036import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
037import gov.nist.secauto.oscal.lib.profile.resolver.support.IEntityItem;
038
039import org.apache.logging.log4j.LogManager;
040import org.apache.logging.log4j.Logger;
041
042import java.util.List;
043import java.util.Locale;
044
045import edu.umd.cs.findbugs.annotations.NonNull;
046
047public class InsertReferencePolicy
048    extends AbstractCustomReferencePolicy<InsertAnchorNode> {
049  private static final Logger LOGGER = LogManager.getLogger(InsertReferencePolicy.class);
050
051  public InsertReferencePolicy() {
052    super(IIdentifierParser.IDENTITY_PARSER);
053  }
054
055  @Override
056  protected List<IEntityItem.ItemType> getEntityItemTypes(@NonNull InsertAnchorNode insert) {
057    String type = insert.getType().toString();
058
059    List<IEntityItem.ItemType> itemTypes;
060    if ("param".equals(type)) {
061      itemTypes = CollectionUtil.singletonList(IEntityItem.ItemType.PARAMETER);
062    } else {
063      throw new UnsupportedOperationException("unrecognized insert type: " + type);
064    }
065    return itemTypes;
066  }
067
068  @Override
069  public String getReferenceText(@NonNull InsertAnchorNode insert) {
070    return insert.getIdReference().toString();
071  }
072
073  @Override
074  public void setReferenceText(@NonNull InsertAnchorNode insert, @NonNull String newReference) {
075    insert.setIdReference(ObjectUtils.notNull(BasedSequence.of(newReference)));
076  }
077
078  @Override
079  protected boolean handleIndexMiss(
080      @NonNull IRequiredValueModelNodeItem contextItem,
081      @NonNull InsertAnchorNode insert,
082      @NonNull List<IEntityItem.ItemType> itemTypes,
083      @NonNull String identifier,
084      @NonNull ReferenceCountingVisitor.Context visitorContext) {
085    if (LOGGER.isErrorEnabled()) {
086      LOGGER.atError().log(
087          "The '{}' insert at '{}' should reference a '{}' identified by '{}'."
088              + " The index did not contain the identifier.",
089          insert.getType().toString(),
090          contextItem.toPath(IPathFormatter.METAPATH_PATH_FORMATER),
091          itemTypes.stream()
092              .map(type -> type.name().toLowerCase(Locale.ROOT))
093              .collect(CustomCollectors.joiningWithOxfordComma("or")),
094          identifier);
095    }
096    return true;
097  }
098
099}
