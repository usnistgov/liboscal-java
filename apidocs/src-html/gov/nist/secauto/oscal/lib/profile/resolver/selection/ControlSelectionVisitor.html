027package gov.nist.secauto.oscal.lib.profile.resolver.selection;
028
029import gov.nist.secauto.metaschema.model.common.metapath.item.IDocumentNodeItem;
030import gov.nist.secauto.metaschema.model.common.metapath.item.IRequiredValueModelNodeItem;
031import gov.nist.secauto.metaschema.model.common.metapath.item.IRootAssemblyNodeItem;
032import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
033import gov.nist.secauto.oscal.lib.model.Catalog;
034import gov.nist.secauto.oscal.lib.model.CatalogGroup;
035import gov.nist.secauto.oscal.lib.model.Control;
036import gov.nist.secauto.oscal.lib.model.ControlPart;
037import gov.nist.secauto.oscal.lib.profile.resolver.support.AbstractIndexingVisitor;
038import gov.nist.secauto.oscal.lib.profile.resolver.support.IEntityItem;
039import gov.nist.secauto.oscal.lib.profile.resolver.support.IIndexer;
040import gov.nist.secauto.oscal.lib.profile.resolver.support.IIndexer.SelectionStatus;
041
042import org.apache.logging.log4j.LogManager;
043import org.apache.logging.log4j.Logger;
044
045import edu.umd.cs.findbugs.annotations.NonNull;
046
047/**
048 * Walks a {@link Catalog} indexing all nodes that can be referenced.
049 * <p>
050 * For each {@link CatalogGroup}, {@link Control}, and {@link ControlPart}, determine if that object
051 * is {@link SelectionStatus#SELECTED} or {@link SelectionStatus#UNSELECTED}.
052 * <p>
053 * A {@link Control} is {@link SelectionStatus#SELECTED} if it matches the configured
054 * {@link IControlFilter}, otherwise it is {@link SelectionStatus#UNSELECTED}.
055 * <p>
056 * A {@link CatalogGroup} is {@link SelectionStatus#SELECTED} if it contains a
057 * {@link SelectionStatus#SELECTED} descendant {@link Control}, otherwise it is
058 * {@link SelectionStatus#UNSELECTED}.
059 * <p>
060 * A {@link ControlPart} is {@link SelectionStatus#SELECTED} if its containing control is
061 * {@link SelectionStatus#SELECTED}.
062 * <p>
063 * All other indexed nodes will have the {@link SelectionStatus#UNKNOWN}, since these nodes require
064 * reference counting to determine if they are to be kept or not.
065 */
066public class ControlSelectionVisitor
067    extends AbstractIndexingVisitor<IControlSelectionState, Boolean> {
068  private static final Logger LOGGER = LogManager.getLogger(ControlSelectionVisitor.class);
069
070  private static final ControlSelectionVisitor SINGLETON = new ControlSelectionVisitor();
071
072  public static ControlSelectionVisitor instance() {
073    return SINGLETON;
074  }
075
076  @Override
077  protected IIndexer getIndexer(IControlSelectionState state) {
078    return state.getIndex();
079  }
080
081  @Override
082  protected Boolean newDefaultResult(IControlSelectionState state) {
083    return false;
084  }
085
086  @Override
087  protected Boolean aggregateResults(Boolean first, Boolean second, IControlSelectionState state) {
088    return first || second;
089  }
090
091  public void visit(@NonNull IDocumentNodeItem catalogDocument, @NonNull IControlSelectionState state) {
092    visitCatalog(catalogDocument, state);
093  }
094
095  public void visitProfile(
096      @NonNull IDocumentNodeItem catalogDocument,
097      @NonNull IDocumentNodeItem profileDocument,
098      @NonNull IControlSelectionState state) {
099    visit(catalogDocument, state);
100
101    IRootAssemblyNodeItem root = profileDocument.getRootAssemblyNodeItem();
102
103    visitMetadata(root, state);
104    visitBackMatter(root, state);
105  }
106
107  @Override
108  public Boolean visitCatalog(IDocumentNodeItem catalogDocument, IControlSelectionState state) {
109    getIndexer(state).setSelectionStatus(catalogDocument, SelectionStatus.SELECTED);
110    return super.visitCatalog(catalogDocument, state);
111  }
112
113  @Override
114  public Boolean visitGroup(IRequiredValueModelNodeItem groupItem, Boolean childSelected,
115      IControlSelectionState state) {
116    super.visitGroup(groupItem, childSelected, state);
117    if (LOGGER.isTraceEnabled()) {
118      CatalogGroup group = (CatalogGroup) groupItem.getValue();
119      LOGGER.atTrace().log("Selecting group '{}'. match={}", group.getId(), childSelected);
120    }
121
122    // these should agree
123    assert state.isSelected(groupItem) == childSelected;
124
125    if (childSelected) {
126      getIndexer(state).setSelectionStatus(groupItem, SelectionStatus.SELECTED);
127    } else {
128      getIndexer(state).setSelectionStatus(groupItem, SelectionStatus.UNSELECTED);
129    }
130
131    handlePartSelection(groupItem, childSelected, state);
132    return childSelected;
133  }
134
135  private void handlePartSelection(
136      @NonNull IRequiredValueModelNodeItem groupOrControlItem,
137      boolean selected,
138      IControlSelectionState state) {
139    if (isVisitedItemType(IEntityItem.ItemType.PART)) {
140      SelectionStatus selectionStatus = selected ? SelectionStatus.SELECTED : SelectionStatus.UNSELECTED;
141
142      IIndexer index = getIndexer(state);
143      CHILD_PART_METAPATH.evaluate(groupOrControlItem).asStream()
144          .map(item -> (IRequiredValueModelNodeItem) item)
145          .forEachOrdered(partItem -> {
146            index.setSelectionStatus(ObjectUtils.requireNonNull(partItem), selectionStatus);
147          });
148    }
149  }
150
151  @Override
152  public Boolean visitControl(
153      IRequiredValueModelNodeItem controlItem,
154      Boolean childResult,
155      IControlSelectionState state) {
156    super.visitControl(controlItem, childResult, state);
157
158    boolean selected = state.isSelected(controlItem);
159    if (selected) {
160      getIndexer(state).setSelectionStatus(controlItem, SelectionStatus.SELECTED);
161    } else {
162      getIndexer(state).setSelectionStatus(controlItem, SelectionStatus.UNSELECTED);
163    }
164
165    handlePartSelection(controlItem, selected, state);
166    return selected;
167  }
168}
