027package gov.nist.secauto.oscal.lib.profile.resolver.policy;
028
029import edu.umd.cs.findbugs.annotations.NonNull;
030
031public interface ICustomReferencePolicy<TYPE> extends IReferencePolicy<TYPE> {
032
033  /**
034   * Get the parser to use to parse an entity identifier from the reference text.
035   *
036   * @return the parser
037   */
038  @NonNull
039  IIdentifierParser getIdentifierParser();
040
041  /**
042   * Retrieve the reference text from the {@code reference} object.
043   *
044   * @param reference
045   *          the reference object
046   * @return the reference text or {@code null} if there is no text
047   */
048  String getReferenceText(@NonNull TYPE reference);
049
050  /**
051   * Update the reference text used in the {@code reference} object.
052   *
053   * @param reference
054   *          the reference object
055   * @param newReferenceText
056   *          the reference text replacement
057   */
058  void setReferenceText(@NonNull TYPE reference, @NonNull String newReferenceText);
059}
