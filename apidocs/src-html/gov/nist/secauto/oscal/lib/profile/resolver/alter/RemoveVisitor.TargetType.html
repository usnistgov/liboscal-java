058public class RemoveVisitor implements ICatalogVisitor<Boolean, RemoveVisitor.Context> {
059  public enum TargetType {
060    PARAM("param", Parameter.class),
061    PROP("prop", Property.class),
062    LINK("link", Link.class),
063    PART("part", ControlPart.class),
064    MAPPING("mapping", Control.Mapping.class),
065    MAP("map", MappingEntry.class);
066
067    @NonNull
068    private static final Map<Class<?>, TargetType> CLASS_TO_TYPE;
069    @NonNull
070    private static final Map<String, TargetType> NAME_TO_TYPE;
071    @NonNull
072    private final String fieldName;
073    @NonNull
074    private final Class<?> clazz;
075
076    static {
077      {
078        Map<Class<?>, TargetType> map = new ConcurrentHashMap<>();
079        for (TargetType type : TargetType.values()) {
080          map.put(type.getClazz(), type);
081        }
082        CLASS_TO_TYPE = CollectionUtil.unmodifiableMap(map);
083      }
084
085      {
086        Map<String, TargetType> map = new ConcurrentHashMap<>();
087        for (TargetType type : TargetType.values()) {
088          map.put(type.fieldName(), type);
089        }
090        NAME_TO_TYPE = CollectionUtil.unmodifiableMap(map);
091      }
092    }
093
094    /**
095     * Get the target type associated with the provided {@code clazz}.
096     *
097     * @param clazz
098     *          the class to identify the target type for
099     * @return the associated target type or {@code null} if the class is not associated with a target
100     *         type
101     */
102    @Nullable
103    public static TargetType forClass(@NonNull Class<?> clazz) {
104      Class<?> target = clazz;
105      TargetType retval;
106      // recurse over parent classes to find a match
107      do {
108        retval = CLASS_TO_TYPE.get(target);
109      } while (retval == null && (target = target.getSuperclass()) != null);
110      return retval;
111    }
112
113    /**
114     * Get the target type associated with the provided field {@code name}.
115     *
116     * @param name
117     *          the field name to identify the target type for
118     * @return the associated target type or {@code null} if the name is not associated with a target
119     *         type
120     */
121    @Nullable
122    public static TargetType forFieldName(@Nullable String name) {
123      return name == null ? null : NAME_TO_TYPE.get(name);
124    }
125
126    TargetType(@NonNull String fieldName, @NonNull Class<?> clazz) {
127      this.fieldName = fieldName;
128      this.clazz = clazz;
129    }
130
131    /**
132     * Get the field name associated with the target type.
133     *
134     * @return the name
135     */
136    public String fieldName() {
137      return fieldName;
138    }
139
140    /**
141     * Get the bound class associated with the target type.
142     *
143     * @return the class
144     */
145    public Class<?> getClazz() {
146      return clazz;
147    }

151  @NonNull
152  private static final RemoveVisitor INSTANCE = new RemoveVisitor();
153
154  private static final Map<TargetType, Set<TargetType>> APPLICABLE_TARGETS;
155
156  static {
157    APPLICABLE_TARGETS = new EnumMap<>(TargetType.class);
158    APPLICABLE_TARGETS.put(TargetType.PARAM, Set.of(TargetType.PROP, TargetType.LINK));
159    APPLICABLE_TARGETS.put(TargetType.PART, Set.of(TargetType.PART, TargetType.PROP, TargetType.LINK));
160    APPLICABLE_TARGETS.put(TargetType.MAPPING, Set.of(TargetType.MAP, TargetType.PROP, TargetType.LINK));
161    APPLICABLE_TARGETS.put(TargetType.MAP, Set.of(TargetType.PROP, TargetType.LINK));
162  }
163
164  private static Set<TargetType> getApplicableTypes(@NonNull TargetType type) {
165    return APPLICABLE_TARGETS.getOrDefault(type, CollectionUtil.emptySet());
166  }
167
168  private static <T> boolean handle(
169      @NonNull TargetType itemType,
170      @NonNull Supplier<? extends Collection<T>> supplier,
171      @Nullable Function<T, Boolean> handler,
172      @NonNull Context context) {
173
174    boolean handleChildren = !Collections.disjoint(context.getTargetItemTypes(), getApplicableTypes(itemType));
175    boolean retval = false;
176    if (context.isMatchingType(itemType)) {
177      // if the item type is applicable, attempt to remove any items
178      Iterator<T> iter = supplier.get().iterator();
179      while (iter.hasNext()) {
180        T item = iter.next();
181
182        if (item == null || context.isApplicableTo(item)) {
183          iter.remove();
184          retval = true;
185          // ignore removed items and their children
186        } else if (handler != null && handleChildren) {
187          // handle child items since they are applicable to the search criteria
188          retval = retval || handler.apply(item);
189        }
190      }
191    } else if (handleChildren && handler != null) {
192      // if the child item type is applicable and there is a handler, iterate over children
193      Iterator<T> iter = supplier.get().iterator();
194      while (iter.hasNext()) {
195        T item = iter.next();
196        if (item != null) {
197          retval = retval || handler.apply(item);
198        }
199      }
200    }
201    return retval;
202  }

204  /**
205   * Apply the remove directive.
206   *
207   * @param control
208   *          the control target
209   * @param objectName
210   *          the name flag of a matching node to remove
211   * @param objectClass
212   *          the class flag of a matching node to remove
213   * @param objectId
214   *          the id flag of a matching node to remove
215   * @param objectNamespace
216   *          the namespace flag of a matching node to remove
217   * @param itemType
218   *          the type of a matching node to remove
219   * @return {@code true} if the modification was made or {@code false} otherwise
220   * @throws ProfileResolutionEvaluationException
221   *           if a processing error occurred during profile resolution
222   */
223  public static boolean remove(
224      @NonNull Control control,
225      @Nullable String objectName,
226      @Nullable String objectClass,
227      @Nullable String objectId,
228      @Nullable String objectNamespace,
229      @Nullable TargetType itemType) {
230    return INSTANCE.visitControl(
231        control,
232        new Context(objectName, objectClass, objectId, objectNamespace, itemType));
233  }

247  @Override
248  public Boolean visitControl(Control control, Context context) {
249    assert context != null;
250
251    // visit params
252    boolean retval = handle(
253        TargetType.PARAM,
254        () -> CollectionUtil.listOrEmpty(control.getParams()),
255        child -> visitParameter(ObjectUtils.notNull(child), context),
256        context);
257
258    // visit props
259    retval = retval || handle(
260        TargetType.PROP,
261        () -> CollectionUtil.listOrEmpty(control.getProps()),
262        null,
263        context);
264
265    // visit links
266    retval = retval || handle(
267        TargetType.LINK,
268        () -> CollectionUtil.listOrEmpty(control.getLinks()),
269        null,
270        context);
271
272    // visit parts
273    retval = retval || handle(
274        TargetType.PART,
275        () -> CollectionUtil.listOrEmpty(control.getParts()),
276        child -> visitPart(child, context),
277        context);
278
279    // visit mappings
280    Control.Mapping mapping = control.getMapping();
281    if (mapping != null) {
282      retval = retval || visitMapping(mapping, context);
283    }
284    return retval;
285  }

390  static final class Context {
391    /**
392     * Types with an "name" flag.
393     */
394    @NonNull
395    private static final Set<TargetType> NAME_TYPES = ObjectUtils.notNull(
396        Set.of(TargetType.PART, TargetType.PROP));
397    /**
398     * Types with an "class" flag.
399     */
400    @NonNull
401    private static final Set<TargetType> CLASS_TYPES = ObjectUtils.notNull(
402        Set.of(TargetType.PARAM, TargetType.PART, TargetType.PROP));
403    /**
404     * Types with an "id" flag.
405     */
406    @NonNull
407    private static final Set<TargetType> ID_TYPES = ObjectUtils.notNull(
408        Set.of(TargetType.PARAM, TargetType.PART));
409    /**
410     * Types with an "ns" flag.
411     */
412    @NonNull
413    private static final Set<TargetType> NAMESPACE_TYPES = ObjectUtils.notNull(
414        Set.of(TargetType.PART, TargetType.PROP));
415
416    @Nullable
417    private final String objectName;
418    @Nullable
419    private final String objectClass;
420    @Nullable
421    private final String objectId;
422    @Nullable
423    private final String objectNamespace;
424    @NonNull
425    private final Set<TargetType> targetItemTypes;
426
427    private static boolean filterTypes(
428        @NonNull Set<TargetType> effectiveTypes,
429        @NonNull String criteria,
430        @NonNull Set<TargetType> allowedTypes,
431        @Nullable String value,
432        @Nullable TargetType itemType) {
433      boolean retval = false;
434      if (value != null) {
435        retval = effectiveTypes.retainAll(allowedTypes);
436        if (itemType != null && !allowedTypes.contains(itemType)) {
437          throw new ProfileResolutionEvaluationException(
438              String.format("%s='%s' is not supported for items of type '%s'",
439                  criteria,
440                  value,
441                  itemType.fieldName()));
442        }
443      }
444      return retval;
445    }
446
447    private Context(
448        @Nullable String objectName,
449        @Nullable String objectClass,
450        @Nullable String objectId,
451        @Nullable String objectNamespace,
452        @Nullable TargetType itemType) {
453
454      // determine the set of effective item types to search for
455      // this helps with short-circuit searching for parts of the graph that cannot match
456      @NonNull Set<TargetType> targetItemTypes = ObjectUtils.notNull(EnumSet.allOf(TargetType.class));
457      filterTypes(targetItemTypes, "by-name", NAME_TYPES, objectName, itemType);
458      filterTypes(targetItemTypes, "by-class", CLASS_TYPES, objectClass, itemType);
459      filterTypes(targetItemTypes, "by-id", ID_TYPES, objectId, itemType);
460      filterTypes(targetItemTypes, "by-ns", NAMESPACE_TYPES, objectNamespace, itemType);
461
462      if (itemType != null) {
463        targetItemTypes.retainAll(Set.of(itemType));
464      }
465
466      if (targetItemTypes.isEmpty()) {
467        throw new ProfileResolutionEvaluationException("The filter matches no available item types");
468      }
469
470      this.objectName = objectName;
471      this.objectClass = objectClass;
472      this.objectId = objectId;
473      this.objectNamespace = objectNamespace;
474      this.targetItemTypes = CollectionUtil.unmodifiableSet(targetItemTypes);
475    }

479      return objectName;

498      return getTargetItemTypes().contains(type);
499    }

503      return objectNamespace;

506    private static boolean checkValue(@Nullable String actual, @Nullable String expected) {
507      return expected == null || expected.equals(actual);
508    }
509
510    public boolean isApplicableTo(@NonNull Object obj) {
511      TargetType objectType = TargetType.forClass(obj.getClass());
512
513      boolean retval = objectType != null && getTargetItemTypes().contains(objectType);
514      if (retval) {
515        assert objectType != null;
516
517        // check other criteria
518        String actualName = null;
519        String actualClass = null;
520        String actualId = null;
521        String actualNamespace = null;
522
523        switch (objectType) {
524        case PARAM: {
525          Parameter param = (Parameter) obj;
526          actualClass = param.getClazz();
527          actualId = param.getId();
528          break;
529        }
530        case PROP: {
531          Property prop = (Property) obj;
532          actualName = prop.getName();
533          actualClass = prop.getClazz();
534          actualNamespace = prop.getNs() == null ? IProperty.OSCAL_NAMESPACE.toString() : prop.getNs().toString();
535          break;
536        }
537        case PART: {
538          ControlPart part = (ControlPart) obj;
539          actualName = part.getName();
540          actualClass = part.getClazz();
541          actualId = part.getId() == null ? null : part.getId().toString();
542          actualNamespace = part.getNs() == null ? IProperty.OSCAL_NAMESPACE.toString() : part.getNs().toString();
543          break;
544        }
545        case LINK:
546        case MAPPING:
547        case MAP:
548          // do nothing
549          break;
550        default:
551          throw new UnsupportedOperationException(objectType.name().toLowerCase(Locale.ROOT));
552        }
553
554        retval = checkValue(actualName, getObjectName())
555            && checkValue(actualClass, getObjectClass())
556            && checkValue(actualId, getObjectId())
557            && checkValue(actualNamespace, getObjectNamespace());
558      }
559      return retval;
560    }
