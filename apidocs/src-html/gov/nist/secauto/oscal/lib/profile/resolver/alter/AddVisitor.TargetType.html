061  public enum TargetType {
062    CONTROL("control", Control.class),
063    PARAM("param", Parameter.class),
064    PART("part", ControlPart.class);
065
066    @NonNull
067    private static final Map<Class<?>, TargetType> CLASS_TO_TYPE;
068    @NonNull
069    private static final Map<String, TargetType> NAME_TO_TYPE;
070    @NonNull
071    private final String fieldName;
072    @NonNull
073    private final Class<?> clazz;
074
075    static {
076      {
077        Map<Class<?>, TargetType> map = new ConcurrentHashMap<>();
078        for (TargetType type : TargetType.values()) {
079          map.put(type.getClazz(), type);
080        }
081        CLASS_TO_TYPE = CollectionUtil.unmodifiableMap(map);
082      }
083
084      {
085        Map<String, TargetType> map = new ConcurrentHashMap<>();
086        for (TargetType type : TargetType.values()) {
087          map.put(type.fieldName(), type);
088        }
089        NAME_TO_TYPE = CollectionUtil.unmodifiableMap(map);
090      }
091    }
092
093    /**
094     * Get the target type associated with the provided {@code clazz}.
095     *
096     * @param clazz
097     *          the class to identify the target type for
098     * @return the associated target type or {@code null} if the class is not associated with a target
099     *         type
100     */
101    @Nullable
102    public static TargetType forClass(@NonNull Class<?> clazz) {
103      Class<?> target = clazz;
104      TargetType retval;
105      // recurse over parent classes to find a match
106      do {
107        retval = CLASS_TO_TYPE.get(target);
108      } while (retval == null && (target = target.getSuperclass()) != null);
109      return retval;
110    }
111
112    /**
113     * Get the target type associated with the provided field {@code name}.
114     *
115     * @param name
116     *          the field name to identify the target type for
117     * @return the associated target type or {@code null} if the name is not associated with a target
118     *         type
119     */
120    @Nullable
121    public static TargetType forFieldName(@Nullable String name) {
122      return name == null ? null : NAME_TO_TYPE.get(name);
123    }
124
125    TargetType(@NonNull String fieldName, @NonNull Class<?> clazz) {
126      this.fieldName = fieldName;
127      this.clazz = clazz;
128    }
129
130    /**
131     * Get the field name associated with the target type.
132     *

134     */

137    }

140     * Get the bound class associated with the target type.

251  /**
252   * If the add applies to the current object, then apply the child objects.
253   * <p>
254   * An add applies if:
255   * <ol>
256   * <li>the {@code targetItem} supports all of the children</li>
257   * <li>the context matches if:
258   * <ul>
259   * <li>the target item's id matches the "by-id"; or</li>
260   * <li>the "by-id" is not defined and the target item is the control matching the target
261   * context</li>
262   * </ul>
263   * </li>
264   * </ol>
265   *
266   * @param <T>
267   *          the type of the {@code targetItem}
268   * @param targetItem
269   *          the current target to process
270   * @param titleConsumer
271   *          a consumer to apply a title to or {@code null} if the object has no title field
272   * @param paramsSupplier
273   *          a supplier for the child {@link Parameter} collection
274   * @param propsSupplier
275   *          a supplier for the child {@link Property} collection
276   * @param linksSupplier
277   *          a supplier for the child {@link Link} collection
278   * @param partsSupplier
279   *          a supplier for the child {@link ControlPart} collection
280   * @param context
281   *          the add context
282   * @return {@code true} if a modification was made or {@code false} otherwise
283   */
284  private static <T> boolean handleCurrent(
285      @NonNull T targetItem,
286      @Nullable Consumer<MarkupLine> titleConsumer,
287      @Nullable Supplier<? extends List<Parameter>> paramsSupplier,
288      @Nullable Supplier<? extends List<Property>> propsSupplier,
289      @Nullable Supplier<? extends List<Link>> linksSupplier,
290      @Nullable Supplier<? extends List<ControlPart>> partsSupplier,
291      @NonNull Context context) {
292    boolean retval = false;
293    Position position = context.getPosition();
294    if (context.appliesTo(targetItem) && !context.isSequenceTargeted(targetItem)) {
295      // the target item is the target of the add
296      MarkupLine newTitle = context.getTitle();
297      if (newTitle != null) {
298        assert titleConsumer != null;
299        titleConsumer.accept(newTitle);
300      }
301
302      handleCollection(position, context.getParams(), paramsSupplier);
303      handleCollection(position, context.getProps(), propsSupplier);
304      handleCollection(position, context.getLinks(), linksSupplier);
305      handleCollection(position, context.getParts(), partsSupplier);
306      retval = true;
307    }
308    return retval;
309  }
310
311  private static <T> void handleCollection(
312      @NonNull Position position,
313      @NonNull List<T> newItems,
314      @Nullable Supplier<? extends List<T>> originalCollectionSupplier) {
315    if (originalCollectionSupplier != null) {
316      List<T> oldItems = originalCollectionSupplier.get();
317      if (!newItems.isEmpty()) {
318        if (Position.STARTING.equals(position)) {
319          oldItems.addAll(0, newItems);
320        } else { // ENDING
321          oldItems.addAll(newItems);
322        }
323      }
324    }
325  }

346  private static <T> boolean handleChild(
347      @NonNull TargetType itemType,
348      @NonNull Supplier<? extends List<T>> originalCollectionSupplier,
349      @NonNull Supplier<? extends List<T>> newItemsSupplier,
350      @Nullable Function<T, Boolean> handler,
351      @NonNull Context context) {
352
353    // determine if this child type can match
354    boolean isItemTypeMatch = context.isMatchingType(itemType);
355
356    Set<TargetType> applicableTypes = getApplicableTypes(itemType);
357    boolean descendChild = handler != null && !Collections.disjoint(context.getTargetItemTypes(), applicableTypes);
358
359    boolean retval = false;
360    if (isItemTypeMatch || descendChild) {
361      // if the item type is applicable, attempt to match by id
362      List<T> collection = originalCollectionSupplier.get();
363      ListIterator<T> iter = collection.listIterator();
364      boolean deferred = false;
365      while (iter.hasNext()) {
366        T item = ObjectUtils.requireNonNull(iter.next());
367
368        if (isItemTypeMatch && context.appliesTo(item) && context.isSequenceTargeted(item)) {
369          // if id match, inject the new items into the collection
370          switch (context.getPosition()) {
371          case AFTER: {
372            newItemsSupplier.get().forEach(add -> iter.add(add));
373            retval = true;
374            break;
375          }
376          case BEFORE: {
377            iter.previous();
378            List<T> adds = newItemsSupplier.get();
379            adds.forEach(add -> iter.add(add));
380            item = iter.next();
381            retval = true;
382            break;
383          }
384          case STARTING:
385          case ENDING:
386            deferred = true;
387            break;
388          default:
389            throw new UnsupportedOperationException(context.getPosition().name().toLowerCase(Locale.ROOT));
390          }
391        }
392
393        if (descendChild) {
394          assert handler != null;
395
396          // handle child items since they are applicable to the search criteria
397          retval = retval || handler.apply(item);
398        }
399      }
400
401      if (deferred) {
402        List<T> newItems = newItemsSupplier.get();
403        if (Position.ENDING.equals(context.getPosition())) {
404          collection.addAll(newItems);
405          retval = true;
406        } else if (Position.STARTING.equals(context.getPosition())) {
407          collection.addAll(0, newItems);
408          retval = true;
409        }
410      }
411    }
412    return retval;
413  }

415  @Override
416  public Boolean visitControl(Control control, Context context) {
417    assert context != null;
418
419    if (control.getParams() == null) {
420      control.setParams(new LinkedList<>());
421    }
422
423    if (control.getProps() == null) {
424      control.setProps(new LinkedList<>());
425    }
426
427    if (control.getLinks() == null) {
428      control.setLinks(new LinkedList<>());
429    }
430
431    if (control.getParts() == null) {
432      control.setParts(new LinkedList<>());
433    }
434
435    boolean retval = handleCurrent(
436        control,
437        title -> control.setTitle(title),
438        () -> control.getParams(),
439        () -> control.getProps(),
440        () -> control.getLinks(),
441        () -> control.getParts(),
442        context);
443
444    // visit params
445    retval = retval || handleChild(
446        TargetType.PARAM,
447        () -> control.getParams(),
448        () -> context.getParams(),
449        child -> visitParameter(ObjectUtils.notNull(child), context),
450        context);
451
452    // visit parts
453    retval = retval || handleChild(
454        TargetType.PART,
455        () -> control.getParts(),
456        () -> context.getParts(),
457        child -> visitPart(child, context),
458        context);
459
460    // visit control children
461    for (Control childControl : CollectionUtil.listOrEmpty(control.getControls())) {
462      Set<TargetType> applicableTypes = getApplicableTypes(TargetType.CONTROL);
463      if (!Collections.disjoint(context.getTargetItemTypes(), applicableTypes)) {
464        retval = retval || visitControl(ObjectUtils.requireNonNull(childControl), context);
465      }
466    }
467    return retval;
468  }

470  @Override
471  public Boolean visitParameter(Parameter parameter, Context context) {
472    assert context != null;
473    if (parameter.getProps() == null) {
474      parameter.setProps(new LinkedList<>());
475    }
476
477    if (parameter.getLinks() == null) {
478      parameter.setLinks(new LinkedList<>());
479    }
480
481    return handleCurrent(
482        parameter,
483        null,
484        null,
485        () -> parameter.getProps(),
486        () -> parameter.getLinks(),
487        null,
488        context);
489  }

504    }
505
506    if (part.getLinks() == null) {
507      part.setLinks(new LinkedList<>());
508    }
509
510    if (part.getParts() == null) {
511      part.setParts(new LinkedList<>());
512    }
513
514    boolean retval = handleCurrent(
515        part,
516        null,
517        null,
518        () -> part.getProps(),
519        () -> part.getLinks(),

522
523    // visit parts
524    retval = retval || handleChild(
525        TargetType.PART,
526        () -> part.getParts(),
527        () -> context.getParts(),
528        child -> visitPart(child, context),
529        context);
530    return retval;

533  static class Context {
534    @NonNull
535    private static final Set<TargetType> TITLE_TYPES = ObjectUtils.notNull(
536        Set.of(TargetType.CONTROL, TargetType.PART));
537    @NonNull
538    private static final Set<TargetType> PARAM_TYPES = ObjectUtils.notNull(
539        Set.of(TargetType.CONTROL, TargetType.PARAM));
540    @NonNull
541    private static final Set<TargetType> PROP_TYPES = ObjectUtils.notNull(
542        Set.of(TargetType.CONTROL, TargetType.PARAM, TargetType.PART));
543    @NonNull
544    private static final Set<TargetType> LINK_TYPES = ObjectUtils.notNull(
545        Set.of(TargetType.CONTROL, TargetType.PARAM, TargetType.PART));
546    @NonNull
547    private static final Set<TargetType> PART_TYPES = ObjectUtils.notNull(
548        Set.of(TargetType.CONTROL, TargetType.PART));
549
550    @NonNull
551    private final Control control;
552    @NonNull
553    private final Position position;
554    @Nullable
555    private final String byId;
556    @Nullable
557    private final MarkupLine title;
558    @NonNull
559    private final List<Parameter> params;
560    @NonNull
561    private final List<Property> props;
562    @NonNull
563    private final List<Link> links;
564    @NonNull
565    private final List<ControlPart> parts;
566    @NonNull
567    private final Set<TargetType> targetItemTypes;
568
569    public Context(
570        @NonNull Control control,
571        @NonNull Position position,
572        @Nullable String byId,
573        @Nullable MarkupLine title,
574        @NonNull List<Parameter> params,
575        @NonNull List<Property> props,
576        @NonNull List<Link> links,
577        @NonNull List<ControlPart> parts) {
578
579      Set<TargetType> targetItemTypes = ObjectUtils.notNull(EnumSet.allOf(TargetType.class));
580
581      List<String> additionObjects = new LinkedList<>();
582
583      boolean sequenceTarget = true;
584      if (title != null) {
585        targetItemTypes.retainAll(TITLE_TYPES);
586        additionObjects.add("title");
587        sequenceTarget = false;
588      }
589
590      if (!params.isEmpty()) {
591        targetItemTypes.retainAll(PARAM_TYPES);
592        additionObjects.add("param");
593      }
594
595      if (!props.isEmpty()) {
596        targetItemTypes.retainAll(PROP_TYPES);
597        additionObjects.add("prop");
598        sequenceTarget = false;
599      }
600
601      if (!links.isEmpty()) {
602        targetItemTypes.retainAll(LINK_TYPES);
603        additionObjects.add("link");
604        sequenceTarget = false;
605      }
606
607      if (!parts.isEmpty()) {
608        targetItemTypes.retainAll(PART_TYPES);
609        additionObjects.add("part");
610      }
611
612      if (Position.BEFORE.equals(position) || Position.AFTER.equals(position)) {
613        if (sequenceTarget) {
614          if (sequenceTarget && !params.isEmpty() && parts.isEmpty()) {
615            targetItemTypes.retainAll(Set.of(TargetType.PARAM));
616          } else if (sequenceTarget && !parts.isEmpty() && params.isEmpty()) {
617            targetItemTypes.retainAll(Set.of(TargetType.PART));
618          } else {
619            throw new ProfileResolutionEvaluationException(
620                "When using position before or after, only one collection of parameters or parts can be specified.");
621          }
622        } else {
623          throw new ProfileResolutionEvaluationException(
624              "When using position before or after, one collection of parameters or parts can be specified."
625                  + " Other additions must not be used.");
626        }
627      }
628
629      if (targetItemTypes.isEmpty()) {
630        throw new ProfileResolutionEvaluationException("No parent object supports the requested objects to add: " +
631            additionObjects.stream().collect(CustomCollectors.joiningWithOxfordComma("or")));
632      }
633
634      this.control = control;
635      this.position = position;
636      this.byId = byId;
637      this.title = title;
638      this.params = params;
639      this.props = props;
640      this.links = links;
641      this.parts = parts;
642      this.targetItemTypes = CollectionUtil.unmodifiableSet(targetItemTypes);
643    }

693    public <T> boolean isSequenceTargeted(T targetItem) {
694      TargetType objectType = TargetType.forClass(targetItem.getClass());
695      return (Position.BEFORE.equals(position) || Position.AFTER.equals(position))
696          && (TargetType.PARAM.equals(objectType) && isMatchingType(TargetType.PARAM)
697              || TargetType.PART.equals(objectType) && isMatchingType(TargetType.PART));
698    }

704    /**
705     * Determine if the provided {@code obj} is the target of the add.
706     *
707     * @param obj
708     *          the current object
709     * @return {@code true} if the current object applies or {@code false} otherwise
710     */
711    public boolean appliesTo(@NonNull Object obj) {
712      TargetType objectType = TargetType.forClass(obj.getClass());
713
714      boolean retval = objectType != null && isMatchingType(objectType);
715      if (retval) {
716        assert objectType != null;
717
718        // check other criteria
719        String actualId = null;
720
721        switch (objectType) {
722        case CONTROL: {
723          Control control = (Control) obj;
724          actualId = control.getId();
725          break;
726        }
727        case PARAM: {
728          Parameter param = (Parameter) obj;
729          actualId = param.getId();
730          break;
731        }
732        case PART: {
733          ControlPart part = (ControlPart) obj;
734          actualId = part.getId() == null ? null : part.getId().toString();
735          break;
736        }
737        default:
738          throw new UnsupportedOperationException(objectType.fieldName());
739        }
740
741        String byId = getById();
