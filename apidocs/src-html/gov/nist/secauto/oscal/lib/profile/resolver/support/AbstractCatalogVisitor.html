027package gov.nist.secauto.oscal.lib.profile.resolver.support;
028
029import gov.nist.secauto.metaschema.model.common.metapath.item.IDocumentNodeItem;
030import gov.nist.secauto.metaschema.model.common.metapath.item.IRequiredValueModelNodeItem;
031import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
032
033import edu.umd.cs.findbugs.annotations.NonNull;
034
035/**
036 * Used to visit a catalog containing groups and controls.
037 *
038 * @param <T>
039 *          the type of the state object used to pass calling context information
040 * @param <R>
041 *          the type of the result for visiting a collection of groups and/or controls
042 */
043public abstract class AbstractCatalogVisitor<T, R> implements ICatalogVisitor<T, R> {
044
045  protected abstract R newDefaultResult(T state);
046
047  protected abstract R aggregateResults(R first, R second, T state);
048
049  protected R visitCatalog(@NonNull IDocumentNodeItem catalogDocument, T state) {
050    return visitGroupContainer(catalogDocument.getRootAssemblyNodeItem(), newDefaultResult(state), state);
051  }
052
053  /**
054   * Visit the child groups and controls (in that order) of a given catalog or group container.
055   *
056   * @param catalogOrGroup
057   *          the catalog or group Metapath item currently being visited
058   * @param initialResult
059   *          the initial result value to use when aggregating child results
060   * @param state
061   *          the current visitor state
062   * @return a meaningful result of the given type
063   */
064  protected R visitGroupContainer(
065      @NonNull IRequiredValueModelNodeItem catalogOrGroup,
066      R initialResult,
067      T state) {
068    R result = catalogOrGroup.getModelItemsByName("group").stream()
069        .map(groupItem -> {
070          return visitGroupItem(ObjectUtils.requireNonNull(groupItem), state);
071        })
072        .reduce(initialResult, (first, second) -> aggregateResults(first, second, state));
073    return visitControlContainer(catalogOrGroup, result, state);
074  }
075
076  /**
077   * Called when visiting a group.
078   * <p>
079   * This method will first visit the group's children, then the group itself.
080   *
081   * @param item
082   *          the group Metapath item to visit
083   * @param state
084   *          the current visitor state
085   * @return a meaningful result of the given type
086   */
087  protected R visitGroupItem(@NonNull IRequiredValueModelNodeItem item, T state) {
088    R childResult = visitGroupContainer(item, newDefaultResult(state), state);
089    return visitGroupInternal(item, childResult, state);
090  }
091
092  /**
093   * Called when visiting a group after visiting it's children.
094   *
095   * @param item
096   *          the group Metapath item currently being visited
097   * @param childResult
098   *          the result of visiting the group's children
099   * @param state
100   *          the current visitor state
101   * @return a meaningful result of the given type
102   */
103  protected R visitGroupInternal(
104      @NonNull IRequiredValueModelNodeItem item,
105      R childResult,
106      T state) {
107    return visitGroup(item, childResult, state);
108  }
109
110  /**
111   * Visit the child controls (in that order) of a given catalog, group, or control container.
112   *
113   * @param catalogOrGroupOrControl
114   *          the catalog, group, or control Metapath item currently being visited
115   * @param initialResult
116   *          the initial result value to use when aggregating child results
117   * @param state
118   *          the current visitor state
119   * @return a meaningful result of the given type
120   */
121  protected R visitControlContainer(
122      @NonNull IRequiredValueModelNodeItem catalogOrGroupOrControl,
123      R initialResult,
124      T state) {
125    return catalogOrGroupOrControl.getModelItemsByName("control").stream()
126        .map(controlItem -> {
127          return visitControlItem(ObjectUtils.requireNonNull(controlItem), state);
128        })
129        .reduce(initialResult, (first, second) -> aggregateResults(first, second, state));
130  }
131
132  /**
133   * Called when visiting a control.
134   * <p>
135   * This method will first visit the control's children, then the control itself.
136   *
137   * @param item
138   *          the group Metapath item to visit
139   * @param state
140   *          the current visitor state
141   * @return a meaningful result of the given type
142   */
143  protected R visitControlItem(@NonNull IRequiredValueModelNodeItem item, T state) {
144    R childResult = visitControlContainer(item, newDefaultResult(state), state);
145    return visitControlInternal(item, childResult, state);
146  }
147
148  /**
149   * Called when visiting a control after visiting it's children.
150   *
151   * @param controlItem
152   *          the Metapath item for the control currently being visited
153   * @param childResult
154   *          the result of visiting the control's children
155   * @param state
156   *          the calling context information
157   * @return a meaningful result of the given type
158   */
159  protected R visitControlInternal(@NonNull IRequiredValueModelNodeItem controlItem, R childResult, T state) {
160    return visitControl(controlItem, childResult, state);
161  }
162}
