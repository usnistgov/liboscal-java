027package gov.nist.secauto.oscal.lib.profile.resolver.support;
028
029import gov.nist.secauto.metaschema.model.common.metapath.item.IRequiredValueModelNodeItem;
030
031import java.net.URI;
032
033import edu.umd.cs.findbugs.annotations.NonNull;
034
035public interface IEntityItem {
036
037  enum ItemType {
038    ROLE(false),
039    LOCATION(true),
040    PARTY(true),
041    GROUP(false),
042    CONTROL(false),
043    PART(false),
044    PARAMETER(false),
045    RESOURCE(true);
046
047    private final boolean uuid;
048
049    ItemType(boolean isUuid) {
050      this.uuid = isUuid;
051    }
052
053    public boolean isUuid() {
054      return uuid;
055    }
056  }
057
058  /**
059   * Get the identifier originally assigned to this entity.
060   * <p>
061   * If the identifier value was reassigned, the return value of this method will
062   * be different than value returned by {@link #getIdentifier()}. In such cases,
063   * a call to {@link #isIdentifierReassigned()} is expected to return
064   * {@code true}.
065   * <p>
066   * If the value was not reassigned, the return value of this method will be the
067   * same value returned by {@link #getIdentifier()}. In this case,
068   * {@link #isIdentifierReassigned()} is expected to return {@code false}.
069   *
070   * @return the original identifier value before reassignment
071   */
072  @NonNull
073  String getOriginalIdentifier();
074
075  /**
076   * Get the entity's current identifier value.
077   *
078   * @return the identifier value
079   */
080  @NonNull
081  String getIdentifier();
082
083  /**
084   * Determine if the identifier was reassigned.
085   *
086   * @return {@code true} if the identifier was reassigned, or {@code false}
087   *         otherwise
088   */
089  boolean isIdentifierReassigned();
090
091  @NonNull
092  IRequiredValueModelNodeItem getInstance();
093
094  void setInstance(@NonNull IRequiredValueModelNodeItem item);
095
096  @NonNull
097  <T> T getInstanceValue();
098
099  @NonNull
100  ItemType getItemType();
101
102  URI getSource();
103
104  int getReferenceCount();
105
106  void incrementReferenceCount();
107
108  int resetReferenceCount();
109}
