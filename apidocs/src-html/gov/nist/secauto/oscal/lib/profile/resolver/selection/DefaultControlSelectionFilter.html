027package gov.nist.secauto.oscal.lib.profile.resolver.selection;
028
029import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
030import gov.nist.secauto.oscal.lib.model.Matching;
031import gov.nist.secauto.oscal.lib.model.control.catalog.IControl;
032import gov.nist.secauto.oscal.lib.model.control.profile.IProfileSelectControlById;
033import gov.nist.secauto.oscal.lib.profile.resolver.ProfileResolutionEvaluationException;
034
035import org.apache.commons.lang3.tuple.Pair;
036import org.apache.logging.log4j.LogManager;
037import org.apache.logging.log4j.Logger;
038
039import java.util.Collections;
040import java.util.List;
041import java.util.Objects;
042import java.util.Set;
043import java.util.regex.Pattern;
044import java.util.stream.Collectors;
045
046import edu.umd.cs.findbugs.annotations.NonNull;
047
048public class DefaultControlSelectionFilter implements IControlSelectionFilter {
049  private static final Logger LOGGER = LogManager.getLogger(DefaultControlSelectionFilter.class);
050
051  @NonNull
052  private final List<Selection> selections;
053
054  /**
055   * Construct a new selection filter based on the provided list of select criteria.
056   *
057   * @param selections
058   *          a list of select criteria
059   */
060  @SuppressWarnings("null")
061  public DefaultControlSelectionFilter(@NonNull List<? extends IProfileSelectControlById> selections) {
062    this.selections = selections.stream()
063        // ignore null entries
064        .filter(Objects::nonNull)
065        // create a selection object for the selection
066        .map(selection -> new Selection(selection))
067        .collect(Collectors.toUnmodifiableList());
068  }
069
070  @NonNull
071  @Override
072  public Pair<Boolean, Boolean> apply(IControl control) {
073    String id = control.getId();
074    if (id == null) {
075      throw new ProfileResolutionEvaluationException("control is missing an identifier");
076    }
077    return match(id);
078  }
079
080  /**
081   * Checks if the provided control identifier matches the criteria defined by this object.
082   *
083   * @param id
084   *          the control identifier to match
085   * @return a {@link Pair} whose first member is {@code true} for a match or {@code false} otherwise,
086   *         and whose second member is {@code true} if the match applies to any child controls or
087   *         {@code false} otherwise
088   */
089  @SuppressWarnings("null")
090  @NonNull
091  protected Pair<Boolean, Boolean> match(String id) {
092    return selections.parallelStream()
093        .map(selection -> selection.match(id))
094        // filter out non-matches
095        .filter(pair -> pair.getLeft())
096        // aggregate matches
097        .reduce((first, second) -> {
098          Pair<Boolean, Boolean> result;
099          if (first.getLeft() || second.getLeft()) {
100            // at least one matches
101            boolean withChild = first.getLeft() && first.getRight() || second.getLeft() && second.getRight();
102            result = Pair.of(true, withChild);
103          } else {
104            result = IControlSelectionFilter.NON_MATCH;
105          }
106          return result;
107        })
108        .orElse(NON_MATCH);
109  }
110
111  @SuppressWarnings("PMD.ImplicitSwitchFallThrough")
112  private static Pattern toPattern(@NonNull Matching matching) {
113    String pattern = ObjectUtils.requireNonNull(matching.getPattern());
114    String regex = pattern.chars().boxed().map(ch -> (char) ch.intValue()).map(ch -> {
115
116      String value;
117      switch (ch) {
118      case '*':
119        value = ".*";
120        break;
121      case '?':
122        value = ".";
123        break;
124      case '.':
125      case '+':
126      case '\\':
127      case '[':
128      case ']':
129      case '{':
130      case '}':
131      case '(':
132      case ')':
133      case '^':
134      case '$':
135        value = "\\" + ch;
136        break;
137      default:
138        value = String.valueOf(ch);
139      }
140      return value;
141    }).collect(Collectors.joining("", "^", "$"));
142
143    if (LOGGER.isTraceEnabled()) {
144      LOGGER.atTrace().log("regex: {}", regex);
145    }
146    return Pattern.compile(regex);
147  }
148
149  private static class Selection {
150
151    private final boolean withChildControls;
152    private final Set<String> identifiers;
153    private final List<Pattern> patterns;
154
155    public Selection(IProfileSelectControlById selection) {
156      // process with-child-controls
157      // default is "no"
158      this.withChildControls = "yes".equals(selection.getWithChildControls());
159
160      // process with-ids
161      List<String> ids = selection.getWithIds();
162      if (ids == null) {
163        ids = Collections.emptyList();
164      }
165      this.identifiers = ids.stream()
166          .filter(Objects::nonNull)
167          .collect(Collectors.toUnmodifiableSet());
168
169      // process with-ids
170      List<Matching> matching = selection.getMatching();
171      if (matching == null) {
172        matching = Collections.emptyList();
173      }
174      this.patterns = matching.stream()
175          .filter(Objects::nonNull)
176          .map(DefaultControlSelectionFilter::toPattern)
177          .collect(Collectors.toUnmodifiableList());
178    }
179
180    public boolean isWithChildControls() {
181      return withChildControls;
182    }
183
184    @NonNull
185    protected Pair<Boolean, Boolean> match(String id) {
186      // first check for direct match
187      boolean result = identifiers.stream().anyMatch(controlIdentifier -> controlIdentifier.equals(id));
188      if (!result) {
189        // next check for pattern match
190        result = patterns.stream().anyMatch(pattern -> pattern.asMatchPredicate().test(id));
191      }
192      return ObjectUtils.notNull(Pair.of(result, isWithChildControls()));
193    }
194  }
195}
