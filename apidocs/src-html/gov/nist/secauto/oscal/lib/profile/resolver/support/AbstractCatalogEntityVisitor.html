042/**
043 * Visits a catalog document and its children as designated.
044 * <p>
045 * This implementation is stateless. The {@code T} parameter can be used to convey state as needed.
046 *
047 * @param <T>
048 *          the state type
049 * @param <R>
050 *          the result type
051 */
052public abstract class AbstractCatalogEntityVisitor<T, R>
053    extends AbstractCatalogVisitor<T, R> {
054  @NonNull
055  public static final MetapathExpression CHILD_PART_METAPATH
056      = MetapathExpression.compile("part|part//part");
057  @NonNull
058  private static final MetapathExpression BACK_MATTER_RESOURCES_METAPATH
059      = MetapathExpression.compile("back-matter/resource");
060  @NonNull
061  private static final Set<IEntityItem.ItemType> GROUP_CONTAINER_TYPES
062      = ObjectUtils.notNull(EnumSet.of(
063          IEntityItem.ItemType.GROUP,
064          IEntityItem.ItemType.CONTROL,
065          IEntityItem.ItemType.PARAMETER,
066          IEntityItem.ItemType.PART));
067  @NonNull
068  private static final Set<IEntityItem.ItemType> CONTROL_CONTAINER_TYPES
069      = ObjectUtils.notNull(EnumSet.of(
070          IEntityItem.ItemType.CONTROL,
071          IEntityItem.ItemType.PARAMETER,
072          IEntityItem.ItemType.PART));
073  @NonNull
074  private final Set<IEntityItem.ItemType> itemTypesToVisit;
075
076  /**
077   * Create a new visitor that will visit the item types identified by {@code itemTypesToVisit}.
078   *
079   * @param itemTypesToVisit
080   *          the item type the visitor will visit
081   */
082  public AbstractCatalogEntityVisitor(@NonNull Set<IEntityItem.ItemType> itemTypesToVisit) {
083    this.itemTypesToVisit = CollectionUtil.unmodifiableSet(itemTypesToVisit);
084  }
085
086  public Set<IEntityItem.ItemType> getItemTypesToVisit() {
087    return CollectionUtil.unmodifiableSet(itemTypesToVisit);
088  }
089
090  protected boolean isVisitedItemType(@NonNull IEntityItem.ItemType type) {
091    return itemTypesToVisit.contains(type);
092  }
093
094  @Override
095  public R visitCatalog(IDocumentNodeItem catalogDocument, T state) {
096    R result = super.visitCatalog(catalogDocument, state);
097
098    IRootAssemblyNodeItem root = catalogDocument.getRootAssemblyNodeItem();
099    visitMetadata(root, state);
100    visitBackMatter(root, state);
101    return result;
102  }
103
104  @Override
105  protected R visitGroupContainer(IRequiredValueModelNodeItem catalogOrGroup, R initialResult, T state) {
106    R retval;
107    if (Collections.disjoint(getItemTypesToVisit(), GROUP_CONTAINER_TYPES)) {
108      retval = initialResult;
109    } else {
110      retval = super.visitGroupContainer(catalogOrGroup, initialResult, state);
111    }
112    return retval;
113  }
114
115  @Override
116  protected R visitControlContainer(IRequiredValueModelNodeItem catalogOrGroupOrControl, R initialResult, T state) {
117    R retval;
118    if (Collections.disjoint(getItemTypesToVisit(), CONTROL_CONTAINER_TYPES)) {
119      retval = initialResult;
120    } else {
121      // first descend to all control container children
122      retval = super.visitControlContainer(catalogOrGroupOrControl, initialResult, state);
123
124      // handle parameters
125      if (isVisitedItemType(IEntityItem.ItemType.PARAMETER)) {
126        retval = catalogOrGroupOrControl.getModelItemsByName("param").stream()
127            .map(paramItem -> {
128              return visitParameter(ObjectUtils.requireNonNull(paramItem), catalogOrGroupOrControl, state);
129            })
130            .reduce(retval, (first, second) -> aggregateResults(first, second, state));
131      } // TODO Auto-generated method stub
132    }
133    return retval;
134  }
135
136  protected void visitParts(@NonNull IRequiredValueModelNodeItem groupOrControlItem, T state) {
137    // handle parts
138    if (isVisitedItemType(IEntityItem.ItemType.PART)) {
139      CHILD_PART_METAPATH.evaluate(groupOrControlItem).asStream()
140          .map(item -> (IRequiredValueModelNodeItem) item)
141          .forEachOrdered(partItem -> {
142            visitPart(ObjectUtils.requireNonNull(partItem), groupOrControlItem, state);
143          });
144    }
145  }
146
147  @Override
148  protected R visitGroupInternal(@NonNull IRequiredValueModelNodeItem item, R childResult, T state) {
149    if (isVisitedItemType(IEntityItem.ItemType.PART)) {
150      visitParts(item, state);
151    }
152
153    R retval = childResult;
154    if (isVisitedItemType(IEntityItem.ItemType.GROUP)) {
155      retval = visitGroup(item, retval, state);
156    }
157    return retval;
158  }
159
160  @Override
161  protected R visitControlInternal(IRequiredValueModelNodeItem item, R childResult, T state) {
162    if (isVisitedItemType(IEntityItem.ItemType.PART)) {
163      visitParts(item, state);
164    }
165
166    R retval = childResult;
167    if (isVisitedItemType(IEntityItem.ItemType.CONTROL)) {
168      retval = visitControl(item, retval, state);
169    }
170    return retval;
171  }

173  /**
174   * Called when visiting a parameter.
175   * <p>
176   * Can be overridden by classes extending this interface to support processing of the visited
177   * object.
178   *
179   * @param item
180   *          the Metapath item for the parameter
181   * @param catalogOrGroupOrControl
182   *          the parameter's parent Metapath item
183   * @param state
184   *          the calling context information
185   * @return a meaningful result of the given type
186   */
187  protected R visitParameter(
188      @NonNull IRequiredValueModelNodeItem item,
189      @NonNull IRequiredValueModelNodeItem catalogOrGroupOrControl,
190      T state) {
191    // do nothing
192    return newDefaultResult(state);
193  }

196   * Called when visiting a part.
197   * <p>
198   * Can be overridden by classes extending this interface to support processing of the visited
199   * object.
200   *
201   * @param item
202   *          the Metapath item for the part
203   * @param groupOrControl
204   *          the part's parent Metapath item
205   * @param state
206   *          the calling context information
207   */
208  protected void visitPart( // NOPMD noop default
209      @NonNull IRequiredValueModelNodeItem item,
210      @NonNull IRequiredValueModelNodeItem groupOrControl,
211      T state) {
212    // do nothing
213  }

215  /**
216   * Called when visiting the "metadata" section of an OSCAL document.
217   * <p>
218   * Visits each contained role, location, and party.
219   *
220   * @param rootItem
221   *          the root Metaschema node item containing the "metadata" node
222   * @param state
223   *          the calling context information
224   */
225  protected void visitMetadata(@NonNull IRootAssemblyNodeItem rootItem, T state) {
226    rootItem.getModelItemsByName("metadata").forEach(metadataItem -> {
227      if (isVisitedItemType(IEntityItem.ItemType.ROLE)) {
228        metadataItem.getModelItemsByName("role").forEach(roleItem -> {
229          visitRole(ObjectUtils.requireNonNull(roleItem), metadataItem, state);
230        });
231      }
232
233      if (isVisitedItemType(IEntityItem.ItemType.LOCATION)) {
234        metadataItem.getModelItemsByName("location").forEach(locationItem -> {
235          visitLocation(ObjectUtils.requireNonNull(locationItem), metadataItem, state);
236        });
237      }
238
239      if (isVisitedItemType(IEntityItem.ItemType.PARTY)) {
240        metadataItem.getModelItemsByName("party").forEach(partyItem -> {
241          visitParty(ObjectUtils.requireNonNull(partyItem), metadataItem, state);
242        });
243      }
244    });
245  }

307  /**
308   * Called when visiting the "back-matter" section of an OSCAL document.
309   * <p>
310   * Visits each contained resource.
311   *
312   * @param rootItem
313   *          the root Metaschema node item containing the "back-matter" node
314   * @param state
315   *          the calling context information
316   */
317  protected void visitBackMatter(@NonNull IRootAssemblyNodeItem rootItem, T state) {
318    if (isVisitedItemType(IEntityItem.ItemType.RESOURCE)) {
319      BACK_MATTER_RESOURCES_METAPATH.evaluate(rootItem).asStream()
320          .map(item -> (IRequiredValueModelNodeItem) item)
321          .forEachOrdered(resourceItem -> {
322            visitResource(ObjectUtils.requireNonNull(resourceItem), rootItem, state);
323          });
324    }
325  }

338   *          the calling context information
339   */
340  protected void visitResource( // NOPMD noop default
341      @NonNull IRequiredValueModelNodeItem item,
342      @NonNull IRootAssemblyNodeItem backMatterItem,
343      T state) {
344    // do nothing
345  }
