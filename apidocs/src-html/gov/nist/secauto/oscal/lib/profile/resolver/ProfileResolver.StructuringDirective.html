182  /**
183   * Resolve the profile to a catalog.
184   *
185   * @param profileDocument
186   *          a {@link IDocumentNodeItem} containing the profile to resolve
187   * @param importHistory
188   *          the import stack for cycle detection
189   * @return the resolved profile
190   * @throws IOException
191   *           if an error occurred while loading the profile or an import
192   * @throws ProfileResolutionException
193   *           if an error occurred while resolving the profile
194   */
195  @NonNull
196  protected IDocumentNodeItem resolveProfile(
197      @NonNull IDocumentNodeItem profileDocument,
198      @NonNull Stack<URI> importHistory) throws IOException, ProfileResolutionException {
199    Catalog resolvedCatalog = new Catalog();
200
201    generateMetadata(resolvedCatalog, profileDocument);
202
203    IIndexer index = resolveImports(resolvedCatalog, profileDocument, importHistory);
204    handleReferences(resolvedCatalog, profileDocument, index);
205    handleMerge(resolvedCatalog, profileDocument, index);
206    handleModify(resolvedCatalog, profileDocument);
207
208    return DefaultNodeItemFactory.instance().newDocumentNodeItem(
209        new RootAssemblyDefinition(
210            ObjectUtils.notNull(
211                (IAssemblyClassBinding) OscalBindingContext.instance().getClassBinding(Catalog.class))),
212        resolvedCatalog,
213        profileDocument.getBaseUri());
214  }

222  @NonNull
223  protected IDocumentNodeItem resolve(@NonNull IDocumentNodeItem profileOrCatalog,
224      @NonNull Stack<URI> importHistory)
225      throws IOException, ProfileResolutionException {
226    Object profileObject = profileOrCatalog.getValue();
227
228    IDocumentNodeItem retval;
229    if (profileObject instanceof Catalog) {
230      // already a catalog
231      retval = profileOrCatalog;
232    } else {
233      // must be a profile
234      retval = resolveProfile(profileOrCatalog, importHistory);
235    }
236    return retval;
237  }

254  private static void generateMetadata(@NonNull Catalog resolvedCatalog, @NonNull IDocumentNodeItem profileDocument) {
255    resolvedCatalog.setUuid(UUID.randomUUID());
256
257    Profile profile = toProfile(profileDocument);
258    Metadata profileMetadata = profile.getMetadata();
259
260    Metadata resolvedMetadata = new Metadata();
261    resolvedMetadata.setTitle(profileMetadata.getTitle());
262
263    if (profileMetadata.getVersion() != null) {
264      resolvedMetadata.setVersion(profileMetadata.getVersion());
265    }
266
267    // metadata.setOscalVersion(OscalUtils.OSCAL_VERSION);
268    resolvedMetadata.setOscalVersion(profileMetadata.getOscalVersion());
269
270    resolvedMetadata.setLastModified(ZonedDateTime.now(ZoneOffset.UTC));
271
272    resolvedMetadata.addProp(AbstractProperty.builder("resolution-tool").value("libOSCAL-Java").build());
273
274    URI profileUri = profileDocument.getDocumentUri();
275    resolvedMetadata.addLink(AbstractLink.builder(profileUri).relation("source-profile").build());
276
277    resolvedCatalog.setMetadata(resolvedMetadata);
278  }

280  @NonNull
281  private IIndexer resolveImports(
282      @NonNull Catalog resolvedCatalog,
283      @NonNull IDocumentNodeItem profileDocument,
284      @NonNull Stack<URI> importHistory)
285      throws IOException, ProfileResolutionException {
286
287    IRootAssemblyNodeItem profileItem = profileDocument.getRootAssemblyNodeItem();
288
289    // first verify there is at least one import
290    List<? extends IRequiredValueModelNodeItem> profileImports = profileItem.getModelItemsByName("import");
291    if (profileImports.isEmpty()) {
292      throw new ProfileResolutionException(String.format("Profile '%s' has no imports", profileItem.getBaseUri()));
293    }
294
295    // now process each import
296    IIndexer retval = new BasicIndexer();
297    for (IRequiredValueModelNodeItem profileImportItem : profileImports) {
298      IIndexer result = resolveImport(
299          ObjectUtils.notNull(profileImportItem),
300          profileDocument,
301          importHistory,
302          resolvedCatalog);
303      retval.append(result);
304    }
305    return retval;
306  }

308  @NonNull
309  protected IIndexer resolveImport(
310      @NonNull IRequiredValueModelNodeItem profileImportItem,
311      @NonNull IDocumentNodeItem profileDocument,
312      @NonNull Stack<URI> importHistory,
313      @NonNull Catalog resolvedCatalog) throws IOException, ProfileResolutionException {
314    ProfileImport profileImport = (ProfileImport) profileImportItem.getValue();
315
316    URI importUri = profileImport.getHref();
317    if (importUri == null) {
318      throw new ProfileResolutionException("profileImport.getHref() must return a non-null URI");
319    }
320
321    if (LOGGER.isDebugEnabled()) {
322      LOGGER.atDebug().log("resolving profile import '{}'", importUri);
323    }
324
325    InputSource source = newImportSource(importUri, profileDocument);
326    URI sourceUri = ObjectUtils.notNull(URI.create(source.getSystemId()));
327
328    // check for import cycle
329    try {
330      requireNonCycle(
331          sourceUri,
332          importHistory);
333    } catch (ImportCycleException ex) {
334      throw new IOException(ex);
335    }
336
337    // track the import in the import history
338    importHistory.push(sourceUri);
339    try {
340      IDocumentNodeItem document = getDynamicContext().getDocumentLoader().loadAsNodeItem(source);
341      IDocumentNodeItem importedCatalog = resolve(document, importHistory);
342
343      // Create a defensive deep copy of the document and associated values, since we will be making
344      // changes to the data.
345      try {
346        importedCatalog = DefaultNodeItemFactory.instance().newDocumentNodeItem(
347            importedCatalog.getRootAssemblyNodeItem().getDefinition(),
348            OscalBindingContext.instance().copyBoundObject(importedCatalog.getValue(), null),
349            importedCatalog.getDocumentUri());
350
351        return new Import(profileDocument, profileImportItem)
352            .resolve(importedCatalog, resolvedCatalog);
353      } catch (BindingException ex) {
354        throw new IOException(ex);
355      }
356    } finally {
357      // pop the resolved catalog from the import history
358      URI poppedUri = ObjectUtils.notNull(importHistory.pop());
359      assert sourceUri.equals(poppedUri);
360    }
361  }

363  @NonNull
364  protected InputSource newImportSource(
365      @NonNull URI importUri,
366      @NonNull IDocumentNodeItem profileDocument) throws IOException {
367
368    // Get the entity resolver to resolve relative references in the profile
369    EntityResolver resolver = getEntityResolver(profileDocument.getDocumentUri());
370
371    InputSource source;
372    if (OscalUtils.isInternalReference(importUri)) {
373      // handle internal reference
374      String uuid = OscalUtils.internalReferenceFragmentToId(importUri);
375
376      IRootAssemblyNodeItem profileItem = profileDocument.getRootAssemblyNodeItem();
377      Profile profile = toProfile(profileItem);
378      Resource resource = profile.getResourceByUuid(ObjectUtils.notNull(UUID.fromString(uuid)));
379      if (resource == null) {
380        throw new IOException(
381            String.format("unable to find the resource identified by '%s' used in profile import", importUri));
382      }
383
384      source = OscalUtils.newInputSource(resource, resolver, null);
385    } else {
386      try {
387        source = resolver.resolveEntity(null, importUri.toASCIIString());
388      } catch (SAXException ex) {
389        throw new IOException(ex);
390      }
391    }
392
393    if (source == null || source.getSystemId() == null) {
394      throw new IOException(String.format("Unable to resolve import '%s'.", importUri.toString()));
395    }
396
397    return source;
398  }

400  private static void requireNonCycle(@NonNull URI uri, @NonNull Stack<URI> importHistory)
401      throws ImportCycleException {
402    List<URI> cycle = checkCycle(uri, importHistory);
403    if (!cycle.isEmpty()) {
404      throw new ImportCycleException(String.format("Importing resource '%s' would result in the import cycle: %s", uri,
405          cycle.stream().map(cycleUri -> cycleUri.toString()).collect(Collectors.joining(" -> ", " -> ", ""))));
406    }
407  }
408
409  @NonNull
410  private static List<URI> checkCycle(@NonNull URI uri, @NonNull Stack<URI> importHistory) {
411    int index = importHistory.indexOf(uri);
412
413    List<URI> retval;
414    if (index == -1) {
415      retval = CollectionUtil.emptyList();
416    } else {
417      retval = CollectionUtil.unmodifiableList(
418          ObjectUtils.notNull(importHistory.subList(0, index + 1)));
419    }
420    return retval;
421  }

423  // TODO: move this to an abstract method on profile
424  private static StructuringDirective getStructuringDirective(Profile profile) {
425    Merge merge = profile.getMerge();
426
427    StructuringDirective retval;
428    if (merge == null) {
429      retval = StructuringDirective.FLAT;
430    } else if (merge.getAsIs() != null && merge.getAsIs()) {
431      retval = StructuringDirective.AS_IS;
432    } else if (merge.getCustom() != null) {
433      retval = StructuringDirective.CUSTOM;
434    } else {
435      retval = StructuringDirective.FLAT;
436    }
437    return retval;
438  }

492  @SuppressWarnings("PMD.ExceptionAsFlowControl") // ok
493  protected void handleModify(@NonNull Catalog resolvedCatalog, @NonNull IDocumentNodeItem profileDocument)
494      throws ProfileResolutionException {
495    IDocumentNodeItem resolvedCatalogDocument = DefaultNodeItemFactory.instance().newDocumentNodeItem(
496        new RootAssemblyDefinition(
497            ObjectUtils.notNull(
498                (IAssemblyClassBinding) OscalBindingContext.instance().getClassBinding(Catalog.class))),
499        resolvedCatalog,
500        profileDocument.getBaseUri());
501
502    try {
503      IIndexer indexer = new BasicIndexer();
504      ControlIndexingVisitor visitor = new ControlIndexingVisitor(
505          ObjectUtils.notNull(EnumSet.of(IEntityItem.ItemType.CONTROL, IEntityItem.ItemType.PARAMETER)));
506      visitor.visitCatalog(resolvedCatalogDocument, indexer);
507
508      METAPATH_SET_PARAMETER.evaluate(profileDocument)
509          .forEach(item -> {
510            IRequiredValueAssemblyNodeItem setParameter = (IRequiredValueAssemblyNodeItem) item;
511            try {
512              handleSetParameter(setParameter, indexer);
513            } catch (ProfileResolutionEvaluationException ex) {
514              throw new ProfileResolutionEvaluationException(
515                  String.format("Unable to apply the set-parameter at '%s'. %s",
516                      setParameter.toPath(IPathFormatter.METAPATH_PATH_FORMATER),
517                      ex.getLocalizedMessage()),
518                  ex);
519            }
520          });
521
522      METAPATH_ALTER.evaluate(profileDocument)
523          .forEach(item -> {
524            handleAlter((IRequiredValueAssemblyNodeItem) item, indexer);
525          });
526    } catch (ProfileResolutionEvaluationException ex) {
527      throw new ProfileResolutionException(ex.getLocalizedMessage(), ex);
528    }
529  }

531  protected void handleSetParameter(IRequiredValueAssemblyNodeItem item, IIndexer indexer) {
532    ProfileSetParameter setParameter = (Modify.ProfileSetParameter) item.getValue();
533    String paramId = ObjectUtils.requireNonNull(setParameter.getParamId());
534    IEntityItem entity = indexer.getEntity(IEntityItem.ItemType.PARAMETER, paramId, false);
535    if (entity == null) {
536      throw new ProfileResolutionEvaluationException(
537          String.format(
538              "The parameter '%s' does not exist in the resolved catalog.",
539              paramId));
540    }
541
542    Parameter param = entity.getInstanceValue();
543
544    // apply the set parameter values
545    param.setClazz(ModifyPhaseUtils.mergeItem(param.getClazz(), setParameter.getClazz()));
546    param.setProps(ModifyPhaseUtils.merge(param.getProps(), setParameter.getProps(),
547        ModifyPhaseUtils.identifierKey(Property::getUuid)));
548    param.setLinks(ModifyPhaseUtils.merge(param.getLinks(), setParameter.getLinks(), ModifyPhaseUtils.identityKey()));
549    param.setLabel(ModifyPhaseUtils.mergeItem(param.getLabel(), setParameter.getLabel()));
550    param.setUsage(ModifyPhaseUtils.mergeItem(param.getUsage(), setParameter.getUsage()));
551    param.setConstraints(
552        ModifyPhaseUtils.merge(param.getConstraints(), setParameter.getConstraints(), ModifyPhaseUtils.identityKey()));
553    param.setGuidelines(
554        ModifyPhaseUtils.merge(param.getGuidelines(), setParameter.getGuidelines(), ModifyPhaseUtils.identityKey()));
555    param.setValues(new LinkedList<>(setParameter.getValues()));
556    param.setSelect(setParameter.getSelect());
557  }

559  protected void handleAlter(IRequiredValueAssemblyNodeItem item, IIndexer indexer) {
560    Modify.Alter alter = (Modify.Alter) item.getValue();
561    String controlId = ObjectUtils.requireNonNull(alter.getControlId());
562    IEntityItem entity = indexer.getEntity(IEntityItem.ItemType.CONTROL, controlId, false);
563    if (entity == null) {
564      throw new ProfileResolutionEvaluationException(
565          String.format(
566              "Unable to apply the alter targeting control '%s' at '%s'."
567                  + " The control does not exist in the resolved catalog.",
568              controlId,
569              item.toPath(IPathFormatter.METAPATH_PATH_FORMATER)));
570    }
571    Control control = entity.getInstanceValue();
572
573    METAPATH_ALTER_REMOVE.evaluate(item)
574        .forEach(nodeItem -> {
575          IRequiredValueNodeItem removeItem = (IRequiredValueNodeItem) nodeItem;
576          Modify.Alter.Remove remove = ObjectUtils.notNull((Modify.Alter.Remove) removeItem.getValue());
577
578          try {
579            if (!RemoveVisitor.remove(
580                control,
581                remove.getByName(),
582                remove.getByClass(),
583                remove.getById(),
584                remove.getByNs(),
585                RemoveVisitor.TargetType.forFieldName(remove.getByItemName()))) {
586              throw new ProfileResolutionEvaluationException(
587                  String.format("The remove did not match a valid target"));
588            }
589          } catch (ProfileResolutionEvaluationException ex) {
590            throw new ProfileResolutionEvaluationException(
591                String.format("Unable to apply the remove targeting control '%s' at '%s'. %s",
592                    control.getId(),
593                    removeItem.toPath(IPathFormatter.METAPATH_PATH_FORMATER),
594                    ex.getLocalizedMessage()),
595                ex);
596          }
597        });
598    METAPATH_ALTER_ADD.evaluate(item)
599        .forEach(nodeItem -> {
600          IRequiredValueNodeItem addItem = (IRequiredValueNodeItem) nodeItem;
601          Modify.Alter.Add add = ObjectUtils.notNull((Modify.Alter.Add) addItem.getValue());
602          String byId = add.getById();
603          try {
604            if (!AddVisitor.add(
605                control,
606                AddVisitor.Position.forName(add.getPosition()),
607                byId,
608                add.getTitle(),
609                CollectionUtil.listOrEmpty(add.getParams()),
610                CollectionUtil.listOrEmpty(add.getProps()),
611                CollectionUtil.listOrEmpty(add.getLinks()),
612                CollectionUtil.listOrEmpty(add.getParts()))) {
613
614              throw new ProfileResolutionEvaluationException(
615                  String.format("The add did not match a valid target"));
616            }
617          } catch (ProfileResolutionEvaluationException ex) {
618            throw new ProfileResolutionEvaluationException(
619                String.format("Unable to apply the add targeting control '%s'%s at '%s'. %s",
620                    control.getId(),
621                    byId == null ? "" : String.format(" having by-id '%s'", byId),
622                    addItem.toPath(IPathFormatter.METAPATH_PATH_FORMATER),
623                    ex.getLocalizedMessage()),
624                ex);
625          }
626        });
627  }

629  private static void handleReferences(@NonNull Catalog resolvedCatalog, @NonNull IDocumentNodeItem profileDocument,
630      @NonNull IIndexer index) {
631
632    BasicIndexer profileIndex = new BasicIndexer();
633
634    new ControlIndexingVisitor(ObjectUtils.notNull(EnumSet.allOf(ItemType.class)))
635        .visitProfile(profileDocument, profileIndex);
636
637    // copy roles, parties, and locations with prop name:keep and any referenced
638    Metadata resolvedMetadata = resolvedCatalog.getMetadata();
639    resolvedMetadata.setRoles(
640        IIndexer.filterDistinct(
641            ObjectUtils.notNull(CollectionUtil.listOrEmpty(resolvedMetadata.getRoles()).stream()),
642            profileIndex.getEntitiesByItemType(IEntityItem.ItemType.ROLE),
643            item -> item.getId())
644            .collect(Collectors.toCollection(LinkedList::new)));
645    resolvedMetadata.setParties(
646        IIndexer.filterDistinct(
647            ObjectUtils.notNull(CollectionUtil.listOrEmpty(resolvedMetadata.getParties()).stream()),
648            profileIndex.getEntitiesByItemType(IEntityItem.ItemType.PARTY),
649            item -> item.getUuid())
650            .collect(Collectors.toCollection(LinkedList::new)));
651    resolvedMetadata.setLocations(
652        IIndexer.filterDistinct(
653            ObjectUtils.notNull(CollectionUtil.listOrEmpty(resolvedMetadata.getLocations()).stream()),
654            profileIndex.getEntitiesByItemType(IEntityItem.ItemType.LOCATION),
655            item -> item.getUuid())
656            .collect(Collectors.toCollection(LinkedList::new)));
657
658    // copy resources
659    BackMatter resolvedBackMatter = resolvedCatalog.getBackMatter();
660    List<Resource> resolvedResources = resolvedBackMatter == null ? CollectionUtil.emptyList()
661        : CollectionUtil.listOrEmpty(resolvedBackMatter.getResources());
662
663    List<Resource> resources = IIndexer.filterDistinct(
664        ObjectUtils.notNull(resolvedResources.stream()),
665        profileIndex.getEntitiesByItemType(IEntityItem.ItemType.RESOURCE),
666        item -> item.getUuid())
667        .collect(Collectors.toCollection(LinkedList::new));
668
669    if (!resources.isEmpty()) {
670      if (resolvedBackMatter == null) {
671        resolvedBackMatter = new BackMatter();
672        resolvedCatalog.setBackMatter(resolvedBackMatter);
673      }
674
675      resolvedBackMatter.setResources(resources);
676    }
677
678    index.append(profileIndex);
679  }

681  private class DocumentEntityResolver implements EntityResolver {
682    @NonNull
683    private final URI documentUri;
684
685    public DocumentEntityResolver(@NonNull URI documentUri) {
686      this.documentUri = documentUri;
687    }
688
689    @NonNull
690    protected URI getDocumentUri() {
691      return documentUri;
692    }
693
694    @Override
695    public InputSource resolveEntity(String publicId, String systemId) throws SAXException, IOException {
696
697      URI resolvedUri = getDocumentUri().resolve(systemId);
698
699      EntityResolver resolver = getDynamicContext().getDocumentLoader().getEntityResolver();
700
701      InputSource retval;
702      if (resolver == null) {
703        retval = new InputSource(resolvedUri.toASCIIString());
704      } else {
705        retval = resolver.resolveEntity(publicId, resolvedUri.toASCIIString());
706      }
707      return retval;
708    }
709
710  }
