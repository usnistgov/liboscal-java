027package gov.nist.secauto.oscal.lib.profile.resolver.selection;
028
029import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
030import gov.nist.secauto.oscal.lib.model.control.catalog.IControl;
031
032import org.apache.commons.lang3.tuple.Pair;
033
034import java.util.Arrays;
035import java.util.Set;
036import java.util.function.Function;
037import java.util.stream.Collectors;
038
039import edu.umd.cs.findbugs.annotations.NonNull;
040
041public interface IControlSelectionFilter extends Function<IControl, Pair<Boolean, Boolean>> {
042
043  @NonNull
044  Pair<Boolean, Boolean> NON_MATCH = ObjectUtils.notNull(Pair.of(false, false));
045  @NonNull
046  Pair<Boolean, Boolean> MATCH = ObjectUtils.notNull(Pair.of(true, true));
047
048  @NonNull
049  IControlSelectionFilter ALL_MATCH = new IControlSelectionFilter() {
050    @Override
051    public Pair<Boolean, Boolean> apply(IControl control) {
052      return MATCH;
053    }
054  };
055
056  @NonNull
057  IControlSelectionFilter NONE_MATCH = new IControlSelectionFilter() {
058    @Override
059    public Pair<Boolean, Boolean> apply(IControl control) {
060      return NON_MATCH;
061    }
062  };
063
064  @NonNull
065  static IControlSelectionFilter matchIds(@NonNull String... identifiers) {
066    return new IControlSelectionFilter() {
067      private Set<String> keys = Arrays.stream(identifiers).collect(Collectors.toUnmodifiableSet());
068
069      @Override
070      public @NonNull Pair<Boolean, Boolean> apply(IControl control) {
071        return ObjectUtils.notNull(Pair.of(keys.contains(control.getId()), false));
072      }
073
074    };
075  }
076
077  /**
078   * Determines if the control is matched by this filter. This method returns a {@link Pair} where the
079   * first member of the pair indicates if the control matches, and the second indicates if the match
080   * applies to child controls as well.
081   *
082   * @param control
083   *          the control to check for a match
084   * @return a pair indicating the status of the match ({@code true} for a match or {@code false}
085   *         otherwise), and if a match applies to child controls
086   */
087  @NonNull
088  @Override
089  Pair<Boolean, Boolean> apply(IControl control);
090}
