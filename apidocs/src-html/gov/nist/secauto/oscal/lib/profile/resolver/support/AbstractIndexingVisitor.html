027package gov.nist.secauto.oscal.lib.profile.resolver.support;
028
029import gov.nist.secauto.metaschema.model.common.metapath.item.IRequiredValueModelNodeItem;
030import gov.nist.secauto.metaschema.model.common.metapath.item.IRootAssemblyNodeItem;
031import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
032
033import java.util.EnumSet;
034import java.util.Set;
035
036import edu.umd.cs.findbugs.annotations.NonNull;
037
038public abstract class AbstractIndexingVisitor<T, R>
039    extends AbstractCatalogEntityVisitor<T, R> {
040
041  public AbstractIndexingVisitor() {
042    this(ObjectUtils.notNull(EnumSet.allOf(IEntityItem.ItemType.class)));
043  }
044
045  public AbstractIndexingVisitor(@NonNull Set<IEntityItem.ItemType> itemTypesToIndex) {
046    super(itemTypesToIndex);
047  }
048
049  @NonNull
050  protected abstract IIndexer getIndexer(T state);
051
052  @Override
053  public R visitGroup(IRequiredValueModelNodeItem item, R childResult, T state) {
054    getIndexer(state).addGroup(item);
055    return childResult;
056  }
057
058  @Override
059  public R visitControl(IRequiredValueModelNodeItem item, R childResult, T state) {
060    getIndexer(state).addControl(item);
061    return childResult;
062  }
063
064  @Override
065  protected R visitParameter(@NonNull IRequiredValueModelNodeItem parameterItem,
066      @NonNull IRequiredValueModelNodeItem catalogOrGroupOrControl, T state) {
067    getIndexer(state).addParameter(parameterItem);
068    return newDefaultResult(state);
069  }
070
071  @Override
072  protected void visitPart(@NonNull IRequiredValueModelNodeItem partItem,
073      @NonNull IRequiredValueModelNodeItem catalogOrGroupOrControl, T state) {
074    getIndexer(state).addPart(partItem);
075  }
076
077  @Override
078  protected void visitRole(IRequiredValueModelNodeItem roleItem, IRequiredValueModelNodeItem metadataItem, T state) {
079    getIndexer(state).addRole(roleItem);
080  }
081
082  @Override
083  protected void visitLocation(IRequiredValueModelNodeItem locationItem, IRequiredValueModelNodeItem metadataItem,
084      T state) {
085    getIndexer(state).addLocation(locationItem);
086  }
087
088  @Override
089  protected void visitParty(IRequiredValueModelNodeItem partyItem, IRequiredValueModelNodeItem metadataItem, T state) {
090    getIndexer(state).addParty(partyItem);
091  }
092
093  @Override
094  protected void visitResource(IRequiredValueModelNodeItem resourceItem, IRootAssemblyNodeItem rootItem, T state) {
095    getIndexer(state).addResource(resourceItem);
096  }
097}
