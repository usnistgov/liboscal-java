018/**
019 * A mapping between two target resources.
020 */
021@MetaschemaAssembly(
022    formalName = "Control Mapping",
023    description = "A mapping between two target resources.",
024    name = "mapping",
025    metaschema = OscalMappingMetaschema.class
026)
027public class Mapping {
028  @BoundFlag(
029      formalName = "Mapping Universally Unique Identifier",
030      description = "A [machine-oriented](https://pages.nist.gov/OSCAL/concepts/identifier-use/#machine-oriented), [globally unique](https://pages.nist.gov/OSCAL/concepts/identifier-use/#globally-unique) identifier with [cross-instance](https://pages.nist.gov/OSCAL/concepts/identifier-use/#cross-instance) scope that can be used to reference this mapping definition elsewhere in this or other OSCAL instances. The locally defined *UUID* of the `mapping` can be used to reference the data item locally or globally (e.g., in an imported OSCAL instance). This UUID should be assigned [per-subject](https://pages.nist.gov/OSCAL/concepts/identifier-use/#consistency), which means it should be consistently used to identify the same mapping across revisions of the document.",
031      useName = "uuid",
032      required = true,
033      typeAdapter = UuidAdapter.class
034  )
035  private UUID _uuid;
036
037  @BoundAssembly(
038      formalName = "Mapped Resource Reference",
039      description = "A reference to a resource that is either the source or target of a mapping.",
040      useName = "source-resource",
041      minOccurs = 1
042  )
043  private MappingResourceReference _sourceResource;
044
045  @BoundAssembly(
046      formalName = "Mapped Resource Reference",
047      description = "A reference to a resource that is either the source or target of a mapping.",
048      useName = "target-resource",
049      minOccurs = 1
050  )
051  private MappingResourceReference _targetResource;
052
053  @BoundAssembly(
054      formalName = "Mapping Entry",
055      description = "A relationship-based mapping between a source and target set consisting of members (i.e., controls, control statements) from the respective source and target.",
056      useName = "map",
057      minOccurs = 1,
058      maxOccurs = -1
059  )
060  @GroupAs(
061      name = "maps",
062      inJson = JsonGroupAsBehavior.LIST
063  )
064  private List<MappingEntry> _maps;
065
066  public Mapping() {
067  }
068
069  public UUID getUuid() {
070    return _uuid;
071  }
072
073  public void setUuid(UUID value) {
074    _uuid = value;
075  }
076
077  public MappingResourceReference getSourceResource() {
078    return _sourceResource;
079  }
080
081  public void setSourceResource(MappingResourceReference value) {
082    _sourceResource = value;
083  }
084
085  public MappingResourceReference getTargetResource() {
086    return _targetResource;
087  }
088
089  public void setTargetResource(MappingResourceReference value) {
090    _targetResource = value;
091  }
092
093  public List<MappingEntry> getMaps() {
094    return _maps;
095  }
096
097  public void setMaps(List<MappingEntry> value) {
098    _maps = value;
099  }
100
101  /**
102   * Add a new {@link MappingEntry} item to the underlying collection.
103   * @param item the item to add
104   * @return {@code true}
105   */
106  public boolean addMap(MappingEntry item) {
107    MappingEntry value = ObjectUtils.requireNonNull(item,"item cannot be null");
108    if (_maps == null) {
109      _maps = new LinkedList<>();
110    }
111    return _maps.add(value);
112  }
113
114  /**
115   * Remove the first matching {@link MappingEntry} item from the underlying collection.
116   * @param item the item to remove
117   * @return {@code true} if the item was removed or {@code false} otherwise
118   */
119  public boolean removeMap(MappingEntry item) {
120    MappingEntry value = ObjectUtils.requireNonNull(item,"item cannot be null");
121    return _maps == null ? false : _maps.remove(value);
122  }
123
124  @Override
125  public String toString() {
126    return new ReflectionToStringBuilder(this, MultilineRecursiveToStringStyle.MULTI_LINE_STYLE).toString();
127  }
128}
