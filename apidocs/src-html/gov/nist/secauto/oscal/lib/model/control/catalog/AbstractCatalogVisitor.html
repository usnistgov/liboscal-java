027package gov.nist.secauto.oscal.lib.model.control.catalog;
028
029import gov.nist.secauto.metaschema.model.common.util.CollectionUtil;
030import gov.nist.secauto.metaschema.model.common.util.ObjectUtils;
031import gov.nist.secauto.oscal.lib.model.Catalog;
032import gov.nist.secauto.oscal.lib.model.CatalogGroup;
033import gov.nist.secauto.oscal.lib.model.Control;
034import gov.nist.secauto.oscal.lib.model.Parameter;
035
036import java.util.Objects;
037
038import edu.umd.cs.findbugs.annotations.NonNull;
039
040public abstract class AbstractCatalogVisitor<RESULT, CONTEXT> implements ICatalogVisitor<RESULT, CONTEXT> {
041
042  /**
043   * Get a new initial/default result for the visitor.
044   *
045   * @return the result
046   */
047  protected abstract RESULT defaultResult();
048
049  /**
050   * Aggregate two results into one.
051   *
052   * @param previous
053   *          the first result
054   * @param current
055   *          the next result
056   * @return the result produced by combining the two results
057   */
058  protected RESULT aggregateResult(RESULT previous, RESULT current) {
059    return current;
060  }
061
062  @Override
063  public RESULT visitCatalog(Catalog catalog, CONTEXT context) {
064    RESULT result = CollectionUtil.listOrEmpty(catalog.getGroups()).stream()
065        .filter(Objects::nonNull)
066        .map(childGroup -> visitGroup(ObjectUtils.notNull(childGroup), context))
067        .reduce(defaultResult(), (previous, current) -> aggregateResult(previous, current));
068    result = CollectionUtil.listOrEmpty(catalog.getControls()).stream()
069        .filter(Objects::nonNull)
070        .map(childControl -> visitControl(ObjectUtils.notNull(childControl), context))
071        .reduce(result, (previous, current) -> aggregateResult(previous, current));
072    return CollectionUtil.listOrEmpty(catalog.getParams()).stream()
073        .filter(Objects::nonNull)
074        .map(childParameter -> visitParameter(ObjectUtils.notNull(childParameter), context))
075        .reduce(result, (previous, current) -> aggregateResult(previous, current));
076  }
077
078  @Override
079  public RESULT visitGroup(@NonNull CatalogGroup group, CONTEXT context) {
080    RESULT result = CollectionUtil.listOrEmpty(group.getGroups()).stream()
081        .filter(Objects::nonNull)
082        .map(childGroup -> visitGroup(ObjectUtils.notNull(childGroup), context))
083        .reduce(defaultResult(), (previous, current) -> aggregateResult(previous, current));
084    result = CollectionUtil.listOrEmpty(group.getControls()).stream()
085        .filter(Objects::nonNull)
086        .map(childControl -> visitControl(ObjectUtils.notNull(childControl), context))
087        .reduce(result, (previous, current) -> aggregateResult(previous, current));
088    return CollectionUtil.listOrEmpty(group.getParams()).stream()
089        .filter(Objects::nonNull)
090        .map(childParameter -> visitParameter(ObjectUtils.notNull(childParameter), context))
091        .reduce(result, (previous, current) -> aggregateResult(previous, current));
092  }
093
094  @Override
095  public RESULT visitControl(Control control, CONTEXT context) {
096    RESULT result = CollectionUtil.listOrEmpty(control.getControls()).stream()
097        .filter(Objects::nonNull)
098        .map(childControl -> visitControl(ObjectUtils.notNull(childControl), context))
099        .reduce(defaultResult(), (previous, current) -> aggregateResult(previous, current));
100    return CollectionUtil.listOrEmpty(control.getParams()).stream()
101        .filter(Objects::nonNull)
102        .map(childParameter -> visitParameter(ObjectUtils.notNull(childParameter), context))
103        .reduce(result, (previous, current) -> aggregateResult(previous, current));
104  }
105
106  @Override
107  public RESULT visitParameter(Parameter parameter, CONTEXT context) {
108    return defaultResult();
109  }
110
111}
