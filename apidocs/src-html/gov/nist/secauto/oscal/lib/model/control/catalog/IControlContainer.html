027package gov.nist.secauto.oscal.lib.model.control.catalog;
028
029import gov.nist.secauto.oscal.lib.model.Control;
030import gov.nist.secauto.oscal.lib.model.Parameter;
031
032import java.util.List;
033import java.util.stream.Stream;
034
035import edu.umd.cs.findbugs.annotations.NonNull;
036
037public interface IControlContainer {
038
039  List<Control> getControls();
040
041  /**
042   * Add a new {@link Control} item to the end of the underlying collection.
043   *
044   * @param item
045   *          the item to add
046   * @return {@code true}
047   */
048  boolean addControl(@NonNull Control item);
049
050  /**
051   * Remove the first matching {@link Control} item from the underlying
052   * collection.
053   *
054   * @param item
055   *          the item to remove
056   * @return {@code true} if the item was removed or {@code false} otherwise
057   */
058  boolean removeControl(@NonNull Control item);
059
060  List<Parameter> getParams();
061
062  /**
063   * Add a new {@link Parameter} item to the underlying collection.
064   *
065   * @param item
066   *          the item to add
067   * @return {@code true}
068   */
069  boolean addParam(@NonNull Parameter item);
070
071  /**
072   * Remove the first matching {@link Parameter} item from the underlying
073   * collection.
074   *
075   * @param item
076   *          the item to remove
077   * @return {@code true} if the item was removed or {@code false} otherwise
078   */
079  boolean removeParam(@NonNull Parameter item);
080
081  /**
082   * Get the parameter identifiers referenced in the object's context, but not by
083   * their child objects.
084   *
085   * @return a stream of identifiers
086   */
087  Stream<String> getReferencedParameterIds();
088}
