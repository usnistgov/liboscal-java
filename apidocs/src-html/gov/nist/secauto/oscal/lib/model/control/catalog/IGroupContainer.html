027package gov.nist.secauto.oscal.lib.model.control.catalog;
028
029import gov.nist.secauto.oscal.lib.model.CatalogGroup;
030
031import java.util.List;
032
033import edu.umd.cs.findbugs.annotations.NonNull;
034
035public interface IGroupContainer extends IControlContainer {
036
037  List<CatalogGroup> getGroups();
038
039  /**
040   * Add a new {@link CatalogGroup} item to the end of the underlying collection.
041   *
042   * @param item
043   *          the item to add
044   * @return {@code true}
045   */
046  boolean addGroup(@NonNull CatalogGroup item);
047
048  /**
049   * Remove the first matching {@link CatalogGroup} item from the underlying collection.
050   *
051   * @param item
052   *          the item to remove
053   * @return {@code true} if the item was removed or {@code false} otherwise
054   */
055  boolean removeGroup(@NonNull CatalogGroup item);
056}
