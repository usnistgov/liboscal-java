27  package gov.nist.secauto.oscal.lib.profile.resolver.support;
28  
29  import gov.nist.secauto.metaschema.model.common.metapath.item.IRequiredValueModelNodeItem;
30  
31  import edu.umd.cs.findbugs.annotations.NonNull;
32  
33  /**
34   * Used to visit a catalog containing groups and controls.
35   *
36   * @param <T>
37   *          the type of the context object used to pass calling context
38   *          information
39   * @param <R>
40   *          the type of the result for visiting a collection of groups and/or
41   *          controls
42   */
43  public interface ICatalogVisitor<T, R> {
44  
45    /**
46     * Called when visiting a group.
47     * <p>
48     * Can be overridden by classes extending this interface to support processing
49     * of the visited object.
50     *
51     * @param item
52     *          the Metapath item for the group
53     * @param childResult
54     *          the result of evaluating the group's children
55     * @param state
56     *          the calling context information
57     * @return a meaningful result of the given type
58     */
59    default R visitGroup(@NonNull IRequiredValueModelNodeItem item, R childResult, T state) {
60      // do nothing by default
61      return childResult;
62    }
63  
64    /**
65     * Called when visiting a control.
66     * <p>
67     * Can be overridden by classes extending this interface to support processing
68     * of the visited object.
69     *
70     * @param item
71     *          the Metapath item for the control
72     * @param childResult
73     *          the result of evaluating the control's children
74     * @param state
75     *          the calling context information
76     * @return a meaningful result of the given type
77     */
78    default R visitControl(@NonNull IRequiredValueModelNodeItem item, R childResult, T state) {
79      // do nothing by default
80      return childResult;
81    }
82  }
