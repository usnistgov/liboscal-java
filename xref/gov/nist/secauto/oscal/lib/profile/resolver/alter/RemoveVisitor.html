57  public class RemoveVisitor implements ICatalogVisitor<Boolean, RemoveVisitor.Context> {
58    public enum TargetType {
59      PARAM("param", Parameter.class),
60      PROP("prop", Property.class),
61      LINK("link", Link.class),
62      PART("part", ControlPart.class);
63  
64      @NonNull
65      private static final Map<Class<?>, TargetType> CLASS_TO_TYPE;
66      @NonNull
67      private static final Map<String, TargetType> NAME_TO_TYPE;
68      @NonNull
69      private final String fieldName;
70      @NonNull
71      private final Class<?> clazz;
72  
73      static {
74        {
75          Map<Class<?>, TargetType> map = new ConcurrentHashMap<>();
76          for (TargetType type : TargetType.values()) {
77            map.put(type.getClazz(), type);
78          }
79          CLASS_TO_TYPE = CollectionUtil.unmodifiableMap(map);
80        }
81  
82        {
83          Map<String, TargetType> map = new ConcurrentHashMap<>();
84          for (TargetType type : TargetType.values()) {
85            map.put(type.fieldName(), type);
86          }
87          NAME_TO_TYPE = CollectionUtil.unmodifiableMap(map);
88        }
89      }
90  
91      /**
92       * Get the target type associated with the provided {@code clazz}.
93       *
94       * @param clazz
95       *          the class to identify the target type for
96       * @return the associated target type or {@code null} if the class is not associated with a target
97       *         type
98       */
99      @Nullable
100     public static TargetType forClass(@NonNull Class<?> clazz) {
101       Class<?> target = clazz;
102       TargetType retval;
103       // recurse over parent classes to find a match
104       do {
105         retval = CLASS_TO_TYPE.get(target);
106       } while (retval == null && (target = target.getSuperclass()) != null);
107       return retval;
108     }
109 
110     /**
111      * Get the target type associated with the provided field {@code name}.
112      *
113      * @param name
114      *          the field name to identify the target type for
115      * @return the associated target type or {@code null} if the name is not associated with a target
116      *         type
117      */
118     @Nullable
119     public static TargetType forFieldName(@Nullable String name) {
120       return name == null ? null : NAME_TO_TYPE.get(name);
121     }
122 
123     TargetType(@NonNull String fieldName, @NonNull Class<?> clazz) {
124       this.fieldName = fieldName;
125       this.clazz = clazz;
126     }
127 
128     /**
129      * Get the field name associated with the target type.
130      *
131      * @return the name
132      */
133     public String fieldName() {
134       return fieldName;
135     }

137     /**
138      * Get the bound class associated with the target type.
139      *
140      * @return the class
141      */
142     public Class<?> getClazz() {
143       return clazz;
144     }

148   @NonNull
149   private static final RemoveVisitor INSTANCE = new RemoveVisitor();
150 
151   private static final Map<TargetType, Set<TargetType>> APPLICABLE_TARGETS;
152 
153   static {
154     APPLICABLE_TARGETS = new EnumMap<>(TargetType.class);
155     APPLICABLE_TARGETS.put(TargetType.PARAM, Set.of(TargetType.PROP, TargetType.LINK));
156     APPLICABLE_TARGETS.put(TargetType.PART, Set.of(TargetType.PART, TargetType.PROP, TargetType.LINK));
157   }
158 
159   private static Set<TargetType> getApplicableTypes(@NonNull TargetType type) {
160     return APPLICABLE_TARGETS.getOrDefault(type, CollectionUtil.emptySet());
161   }
162 
163   private static <T> boolean handle(
164       @NonNull TargetType itemType,
165       @NonNull Supplier<? extends Collection<T>> supplier,
166       @Nullable Function<T, Boolean> handler,
167       @NonNull Context context) {
168 
169     boolean handleChildren = !Collections.disjoint(context.getTargetItemTypes(), getApplicableTypes(itemType));
170     boolean retval = false;
171     if (context.isMatchingType(itemType)) {
172       // if the item type is applicable, attempt to remove any items
173       Iterator<T> iter = supplier.get().iterator();
174       while (iter.hasNext()) {
175         T item = iter.next();
176 
177         if (item == null || context.isApplicableTo(item)) {
178           iter.remove();
179           retval = true;
180           // ignore removed items and their children
181         } else if (handler != null && handleChildren) {
182           // handle child items since they are applicable to the search criteria
183           retval = retval || handler.apply(item);
184         }
185       }
186     } else if (handleChildren && handler != null) {
187       // if the child item type is applicable and there is a handler, iterate over children
188       Iterator<T> iter = supplier.get().iterator();
189       while (iter.hasNext()) {
190         T item = iter.next();
191         if (item != null) {
192           retval = retval || handler.apply(item);
193         }
194       }
195     }
196     return retval;
197   }

199   /**
200    * Apply the remove directive.
201    *
202    * @param control
203    *          the control target
204    * @param objectName
205    *          the name flag of a matching node to remove
206    * @param objectClass
207    *          the class flag of a matching node to remove
208    * @param objectId
209    *          the id flag of a matching node to remove
210    * @param objectNamespace
211    *          the namespace flag of a matching node to remove
212    * @param itemType
213    *          the type of a matching node to remove
214    * @return {@code true} if the modification was made or {@code false} otherwise
215    * @throws ProfileResolutionEvaluationException
216    *           if a processing error occurred during profile resolution
217    */
218   public static boolean remove(
219       @NonNull Control control,
220       @Nullable String objectName,
221       @Nullable String objectClass,
222       @Nullable String objectId,
223       @Nullable String objectNamespace,
224       @Nullable TargetType itemType) {
225     return INSTANCE.visitControl(
226         control,
227         new Context(objectName, objectClass, objectId, objectNamespace, itemType));
228   }

242   @Override
243   public Boolean visitControl(Control control, Context context) {
244     assert context != null;
245 
246     // visit params
247     boolean retval = handle(
248         TargetType.PARAM,
249         () -> CollectionUtil.listOrEmpty(control.getParams()),
250         child -> visitParameter(ObjectUtils.notNull(child), context),
251         context);
252 
253     // visit props
254     retval = retval || handle(
255         TargetType.PROP,
256         () -> CollectionUtil.listOrEmpty(control.getProps()),
257         null,
258         context);
259 
260     // visit links
261     retval = retval || handle(
262         TargetType.LINK,
263         () -> CollectionUtil.listOrEmpty(control.getLinks()),
264         null,
265         context);
266 
267     // visit parts
268     retval = retval || handle(
269         TargetType.PART,
270         () -> CollectionUtil.listOrEmpty(control.getParts()),
271         child -> visitPart(child, context),
272         context);
273 
274     return retval;
275   }

279     assert context != null;
280 
281     // visit props

283         TargetType.PROP,

285         null,
286         context);
287 
288     // visit links

292         null,
293         context);
294     return retval;

297   /**
298    * Visit the control part.
299    *
300    * @param part
301    *          the bound part object
302    * @param context
303    *          the visitor context
304    * @return {@code true} if the removal was applied or {@code false} otherwise
305    */
306   public boolean visitPart(ControlPart part, Context context) {
307     assert context != null;
308 
309     // visit props
310     boolean retval = handle(
311         TargetType.PROP,
312         () -> CollectionUtil.listOrEmpty(part.getProps()),
313         null,
314         context);
315 
316     // visit links
317     retval = retval || handle(
318         TargetType.LINK,
319         () -> CollectionUtil.listOrEmpty(part.getLinks()),
320         null,
321         context);
322 
323     // visit parts
324     retval = retval || handle(
325         TargetType.PART,
326         () -> CollectionUtil.listOrEmpty(part.getParts()),
327         child -> visitPart(child, context),
328         context);
329     return retval;
330   }

332   static final class Context {
333     /**
334      * Types with an "name" flag.
335      */
336     @NonNull
337     private static final Set<TargetType> NAME_TYPES = ObjectUtils.notNull(
338         Set.of(TargetType.PART, TargetType.PROP));
339     /**
340      * Types with an "class" flag.
341      */
342     @NonNull
343     private static final Set<TargetType> CLASS_TYPES = ObjectUtils.notNull(
344         Set.of(TargetType.PARAM, TargetType.PART, TargetType.PROP));
345     /**
346      * Types with an "id" flag.
347      */
348     @NonNull
349     private static final Set<TargetType> ID_TYPES = ObjectUtils.notNull(
350         Set.of(TargetType.PARAM, TargetType.PART));
351     /**
352      * Types with an "ns" flag.
353      */
354     @NonNull
355     private static final Set<TargetType> NAMESPACE_TYPES = ObjectUtils.notNull(
356         Set.of(TargetType.PART, TargetType.PROP));
357 
358     @Nullable
359     private final String objectName;
360     @Nullable
361     private final String objectClass;
362     @Nullable
363     private final String objectId;
364     @Nullable
365     private final String objectNamespace;
366     @NonNull
367     private final Set<TargetType> targetItemTypes;
368 
369     private static boolean filterTypes(
370         @NonNull Set<TargetType> effectiveTypes,
371         @NonNull String criteria,
372         @NonNull Set<TargetType> allowedTypes,
373         @Nullable String value,
374         @Nullable TargetType itemType) {
375       boolean retval = false;
376       if (value != null) {
377         retval = effectiveTypes.retainAll(allowedTypes);
378         if (itemType != null && !allowedTypes.contains(itemType)) {
379           throw new ProfileResolutionEvaluationException(
380               String.format("%s='%s' is not supported for items of type '%s'",
381                   criteria,
382                   value,
383                   itemType.fieldName()));
384         }
385       }
386       return retval;
387     }
388 
389     private Context(
390         @Nullable String objectName,
391         @Nullable String objectClass,
392         @Nullable String objectId,
393         @Nullable String objectNamespace,
394         @Nullable TargetType itemType) {
395 
396       // determine the set of effective item types to search for
397       // this helps with short-circuit searching for parts of the graph that cannot match
398       @NonNull Set<TargetType> targetItemTypes = ObjectUtils.notNull(EnumSet.allOf(TargetType.class));
399       filterTypes(targetItemTypes, "by-name", NAME_TYPES, objectName, itemType);
400       filterTypes(targetItemTypes, "by-class", CLASS_TYPES, objectClass, itemType);
401       filterTypes(targetItemTypes, "by-id", ID_TYPES, objectId, itemType);
402       filterTypes(targetItemTypes, "by-ns", NAMESPACE_TYPES, objectNamespace, itemType);
403 
404       if (itemType != null) {
405         targetItemTypes.retainAll(Set.of(itemType));
406       }
407 
408       if (targetItemTypes.isEmpty()) {
409         throw new ProfileResolutionEvaluationException("The filter matches no available item types");
410       }
411 
412       this.objectName = objectName;
413       this.objectClass = objectClass;
414       this.objectId = objectId;
415       this.objectNamespace = objectNamespace;
416       this.targetItemTypes = CollectionUtil.unmodifiableSet(targetItemTypes);
417     }

448     private static boolean checkValue(@Nullable String actual, @Nullable String expected) {
449       return expected == null || expected.equals(actual);
450     }
451 
452     public boolean isApplicableTo(@NonNull Object obj) {
453       TargetType objectType = TargetType.forClass(obj.getClass());
454 
455       boolean retval = objectType != null && getTargetItemTypes().contains(objectType);
456       if (retval) {
457         assert objectType != null;
458 
459         // check other criteria
460         String actualName = null;
461         String actualClass = null;
462         String actualId = null;
463         String actualNamespace = null;
464 
465         switch (objectType) {
466         case PARAM: {
467           Parameter param = (Parameter) obj;
468           actualClass = param.getClazz();
469           actualId = param.getId();
470           break;
471         }
472         case PROP: {
473           Property prop = (Property) obj;
474           actualName = prop.getName();
475           actualClass = prop.getClazz();
476           actualNamespace = prop.getNs() == null ? IProperty.OSCAL_NAMESPACE.toString() : prop.getNs().toString();
477           break;
478         }
479         case PART: {
480           ControlPart part = (ControlPart) obj;
481           actualName = part.getName();
482           actualClass = part.getClazz();
483           actualId = part.getId() == null ? null : part.getId().toString();
484           actualNamespace = part.getNs() == null ? IProperty.OSCAL_NAMESPACE.toString() : part.getNs().toString();
485           break;
486         }
487         case LINK:
488           // do nothing
489           break;
490         default:
491           throw new UnsupportedOperationException(objectType.name().toLowerCase(Locale.ROOT));
492         }
493 
494         retval = checkValue(actualName, getObjectName())
495             && checkValue(actualClass, getObjectClass())
496             && checkValue(actualId, getObjectId())
497             && checkValue(actualNamespace, getObjectNamespace());
498       }
499       return retval;
500     }
