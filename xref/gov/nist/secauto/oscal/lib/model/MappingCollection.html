19   */
20  @MetaschemaAssembly(
21      formalName = "Mapping Collection",
22      description = "A collection of relationship-based control and/or control statement mappings.",
23      name = "mapping-collection",
24      metaschema = OscalMappingMetaschema.class,
25      rootName = "mapping-collection",
26      remarks = "A mapping collection affirmatively declares the relationships that exist between sets of controls and/or control statements in a source and target. It is expected that inferences can be made based on what is mapped; however, no inferences should be made based on what is not mapped, since it is impossible to quantify how complete or granular a given mapping is."
27  )
28  public class MappingCollection {
29    @BoundFlag(
30        formalName = "Mapping Collection Universally Unique Identifier",
31        description = "A globally unique identifier with cross-instance scope for this catalog instance. This UUID should be changed when this document is revised.",
32        useName = "uuid",
33        required = true,
34        typeAdapter = UuidAdapter.class
35    )
36    private UUID _uuid;

38    @BoundAssembly(
39        formalName = "Document Metadata",
40        description = "Provides information about the containing document, and defines concepts that are shared across the document.",
41        useName = "metadata",
42        minOccurs = 1
43    )
44    private Metadata _metadata;
45  
46    @BoundAssembly(
47        formalName = "Control Mapping",
48        description = "A mapping between two target resources.",
49        useName = "mapping",
50        minOccurs = 1,
51        maxOccurs = -1
52    )
53    @GroupAs(
54        name = "mappings"
55    )

93    /**
94     * Add a new {@link Mapping} item to the underlying collection.
95     * @param item the item to add
96     * @return {@code true}
97     */
98    public boolean addMapping(Mapping item) {
99      Mapping value = ObjectUtils.requireNonNull(item,"item cannot be null");
100     if (_mappings == null) {
101       _mappings = new LinkedList<>();
102     }
103     return _mappings.add(value);
104   }
105 
106   /**
107    * Remove the first matching {@link Mapping} item from the underlying collection.
108    * @param item the item to remove
109    * @return {@code true} if the item was removed or {@code false} otherwise
110    */
111   public boolean removeMapping(Mapping item) {
112     Mapping value = ObjectUtils.requireNonNull(item,"item cannot be null");
113     return _mappings == null ? false : _mappings.remove(value);
114   }
